  3 /** @class
  4 
  5   Templates provide inheritance without any classes.
  6 
  7   The idiomatic use of templates is to have generic
  8   object that uses internal slots to determine what
  9   to do.
 10 
 11   In Object Oriented design, a constructor would look
 12   like the following:
 13 
 14       var Person = Espresso.Class.extend({
 15         init: function (name) {
 16           this.set('name', name);
 17         }
 18       });
 19 
 20       var hoban = new Person("Hoban 'Wash' Washburne");
 21 
 22   Using templates, the equivalent code would be:
 23 
 24       var Person = Espresso.Template.extend({
 25         name: null // Imply a name
 26       });
 27 
 28       var hoban = Person.extend({
 29         name: "Hoban 'Wash' Washburne"
 30       });
 31 
 32   The end result is straightforward (and transparent)
 33   object construction.
 34 
 35   The other benefit to using templates over classes
 36   is the fact that _everything_ is an object, and
 37   therefore can be extended upon. This allows one-off
 38   objects to be created from derivative objects, and
 39   truly encapsulates the concept prototypal inheritance.
 40 
 41   Templates are slightly more advanced than Classes,
 42   since they are typeless and require stateless objects
 43   to perform at their best.
 44 
 45   @extends Espresso.PubSub
 46   @extends Espresso.KVO
 47  */
 48 Espresso.Template = mix(Espresso.PubSub, Espresso.KVO, /** @lends Espresso.Template# */{
 49 
 50   /**
 51     Override `init` to act like a constructor like so:
 52 
 53         var shotgun = Espresso.Template.extend({
 54           init: function () { alert("bang!"); }
 55         });
 56 
 57     These constructors will take no arguments,
 58     and are called after the extending is finished.
 59     For stacked Espresso.Templates, use `around()` to
 60     get the super object passed in as the first argument.
 61 
 62     @returns {void}
 63    */
 64   init: function () {},
 65 
 66   /**
 67     Extend a Template with a collection of objects.
 68 
 69     If you use around to get the super argument of the
 70     base object's function, the function will be augmented
 71     in such a way that you don't have to set the scope in
 72     which the function should be called in. Just call the
 73     function normally, assuming that it is "special" and
 74     will have `this` reference the current context you're in.
 75     If you want to, you certainly have the option to apply
 76     the scope if you want.
 77 
 78     @returns {Espresso.Template} The extended template.
 79 
 80     @example
 81       var Animal = Espresso.Template.extend({
 82         move: function (meters) {
 83           return "{} moved {} m.".fmt(this.name, meters);
 84         }
 85       });
 86 
 87       var Snake = Animal.extend({
 88         move: function ($super) {
 89           return "Slithering... {}".fmt($super(5));
 90         }.around()
 91       });
 92 
 93       var Horse = Animal.extend({
 94         move: function ($super) {
 95           return "Galloping... {}".fmt($super(45));
 96        }.around()
 97       });
 98 
 99       var sam = Snake.extend({ name: "Sammy the Python" });
100       var tom = Horse.extend({ name: "Tommy the Palomino" });
101 
102       alert(sam.move());
103       // -> "Slithering... Sammy the Python moved 5 m."
104       alert(tom.move());
105       // -> "Galloping... Tommy the Palomino moved 45 m."
106    */
107   extend: function () {
108     var F = function () {},
109         extension;
110 
111     F.prototype = this;
112     extension = new F();
113     mix.apply(null, arguments).into(extension);
114 
115     if (Espresso.isCallable(extension.init)) {
116       extension.init();
117     }
118     return extension;
119   },
120 
121   /** @function
122 
123     Filters out private variables, functions, and
124     {@link KVO#unknownProperty} when parsing to JSON.
125 
126     It _will_ get any computed properties (via the
127     `property` decorator on an Object.
128 
129     @returns Object The slots to stringify to JSON.
130    */
131   toJSON: function (key) {
132     var k, v, json = {};
133     for (k in this) {
134       v = this.get(k);
135       if (k.charAt(0) !== "_" && !Espresso.isCallable(v) && k !== 'unknownProperty') {
136         json[k] = v;
137       }
138     }
139 
140     return json;
141   }
142 
143 }).into({});
