1 mix(/** @scope Espresso */{
  2 
  3   /**
  4     Marks a function as a computed property, where the
  5     getter and setter functions are the same function.
  6 
  7     @param {Function} fn The function to be called when
  8       the property should be computed.
  9     @param {...} dependentKeys The dependent keys that
 10       this property has. When any of these keys get
 11       updated via KVO, the property will be notified.
 12     @returns {Espresso.Property} The function as a Espresso.property.
 13    */
 14   property: function (fn, dependentKeys) {
 15     if (Espresso.isCallable(fn)) {
 16       mix(Espresso.Property, {
 17         dependentKeys: Espresso.A(arguments).slice(1)
 18       }).into(fn);
 19     } else {
 20       fn = {};
 21     }
 22 
 23     // Decorator API
 24     fn._ = fn._ || {};
 25     /** @ignore */
 26     fn._.property = function (template, value, key) {
 27       var kvoKey = key,
 28           isComputed = Espresso.isCallable(value),
 29           meta = template.__espmeta__ || {};
 30 
 31       // ECMAScript5 compatible API (no need for get or set!)
 32       try { // IE burps on this
 33         if ("defineProperty" in Object) {
 34           kvoKey = "__kvo__{}__".format(key);
 35 
 36           if (meta[kvoKey]) { delete template[key]; }
 37 
 38           if (isComputed) {
 39             template[kvoKey] = value;
 40             meta[key] = { closureKey: kvoKey };
 41           }
 42 
 43           Object.defineProperty(template, key, {
 44             get: function () { return this.get(kvoKey); },
 45             set: function (value) { return this.set(kvoKey, value); },
 46             enumerable: true,
 47             configurable: true
 48           });
 49 
 50           kvoKey.name = key;
 51           value = void(0);
 52         }
 53       } catch (e) {}
 54 
 55       meta[kvoKey] = { referenceKey: key,
 56                        isComputed: isComputed };
 57       template.__espmeta__ = meta;
 58 
 59       return value;
 60     };
 61 
 62     return fn;
 63   }
 64 
 65 }).into(Espresso);
