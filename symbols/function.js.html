  3 mix(/** @lends Function.prototype */{
  4 
  5   /** @function
  6     @desc
  7     Bind the value of `this` on a function before hand,
  8     with any extra arguments being passed in as initial
  9     arguments.
 10 
 11     This implementation conforms to the ECMAScript 5
 12     standard.
 13 
 14         var barista = function (tpl) {
 15           alert(tpl.format(this));
 16           return arguments.callee.bind(this, "Order up! Your {} is ready!");
 17         };
 18 
 19         orderUp = barista.call("espresso", "I would like an {}");
 20         // -> "I would like an espresso."
 21 
 22         orderUp();
 23         // -> "Order up! Your espresso is ready!"
 24 
 25     @param {Object} thisArg The value to bind `this` to on the function.
 26     @returns {Function} The function passed in, wrapped to ensure `this`
 27       is the correct scope.
 28    */
 29   bind: Espresso.inferior(function (self) {
 30     var Target, A;
 31 
 32     // 1. Let Target be the this value.
 33     Target = this;
 34 
 35     // 2. If IsCallable(Target) is false, throw a TypeError exception
 36     if (!Espresso.isCallable(Target)) {
 37       throw new TypeError("The Target is not callable.");
 38     }
 39 
 40     // 3. Let A be a new (possibly empty) internal list of
 41     //    all argument values provided after self
 42     //    (arg1, arg2, etc), in order
 43     A = Espresso.A(arguments).slice(1);
 44 
 45     var bound = function () {
 46 
 47       if (this instanceof bound) {
 48         // 15.3.4.5.2 [[Construct]]
 49         // When the [[Construct]] internal method of a function object, F,
 50         // that was created using the bind function is called with a list of
 51         // arguments ExtraArgs, the following steps are taken:
 52 
 53         // 1. Let the target be the value of F's [[TargetFunction]] internal property.
 54         // 2. If target has no [[Construct]] internal method, a TypeError exception is thrown.
 55         // 3. Let boundArgs be the value of F's [[BoundArgs]] internal property.
 56         // 4. Left args be a new list containing the same values as the list boundArgs in the same order followed by the same values as the list ExtraArgs in the same order.
 57         // 5. Return the result of calling the [[Construct]] internal method of target providing args as the arguments.
 58         var Type = function () {}, that;
 59         Type.prototype = Target.prototype;
 60         that = new Type();
 61 
 62         Target.apply(that, A.concat(Espresso.A(arguments)));
 63         return that;
 64       } else {
 65         // 15.3.4.5.1 [[Call]]
 66         // When the [[Call]] internal method of a function object, F,
 67         // which was created using the bind function is called with a this
 68         // value and a list of arguments ExtraArgs, the following steps are taken:
 69         // 1. Let boundArgs be the value of F's [[BoundArgs]] internal property.
 70         // 2. Let boundThis be the value of F's [[BoundThis]] internal property.
 71         // 3. Let target be the value of F's [[TargetFunction]] internal property.
 72         return Target.apply(self, A.concat(Espresso.A(arguments)));
 73       }
 74     };
 75     return bound;
 76   })
 77 
 78 }).into(Function.prototype);
