  3 mix(/** @lends Number# */{
  4 
  5   /**
  6     Shim for `toJSON`. Returns the `valueOf` the Number.
  7 
  8     @returns {String} This.
  9    */
 10   toJSON: function (key) {
 11     return this.valueOf();
 12   }.inferior(),
 13 
 14   /**
 15     Formatter for `Number`s.
 16 
 17     Don't call this function- It's here for `Espresso.fmt`
 18     to take care of buisiness for you.
 19 
 20     @param {String} spec The specifier to format the number as.
 21     @returns {String} The number formatted as specified.
 22    */
 23   __fmt__: function (spec) {
 24     // Don't want Infinity, -Infinity and NaN in here!
 25     if (!isFinite(this)) {
 26       return this;
 27     }
 28 
 29     var match = spec.match(Espresso.FMT_SPECIFIER),
 30         align = match[1],
 31         fill = match[2],
 32         sign = match[3] || '-',
 33         base = !!match[4],
 34         minWidth = match[6] || 0,
 35         maxWidth = match[7],
 36         type = match[8], value = this, precision;
 37 
 38     if (align) {
 39       align = align.slice(-1);
 40     }
 41 
 42     if (!fill && !!match[5]) {
 43       fill = '0';
 44       if (!align) {
 45         align = '=';
 46       }
 47     }
 48 
 49     if (maxWidth) {
 50       precision = +maxWidth.slice(1);
 51     }
 52 
 53     switch (sign) {
 54     case '+':
 55       sign = (value >= 0) ? '+': '-';
 56       break;
 57     case '-':
 58       sign = (value >= 0) ? '': '-';
 59       break;
 60     case ' ':
 61       sign = (value >= 0) ? ' ': '-';
 62       break;
 63     default:
 64       sign = "";
 65     }
 66 
 67     if (precision && !isNaN(precision)) {
 68       // Opting to go with a more intuitive approach than Python...
 69       //  >>> "{.2}".format(math.pi)
 70       //  "3.1"
 71       // Which is waaay less intuitive than
 72       //  > "{.2}".fmt(Math.PI)
 73       //  "3.14"
 74       value = +value.toFixed(precision);
 75       precision++; // make floating point precision work like Python.
 76     } else {
 77       precision = null;
 78     }
 79 
 80     value = Math.abs(value);
 81 
 82     switch (type) {
 83     case 'b':
 84       base = base ? '0b': '';
 85       value = base + value.toString(2);
 86       break;
 87     case 'c':
 88       value = String.fromCharCode(value);
 89       break;
 90     case 'o':
 91       base = base ? '0o': '';
 92       value = base + value.toString(8);
 93       break;
 94     case 'x':
 95       base = base ? '0x': '';
 96       value = base + value.toString(16).toLowerCase();
 97       break;
 98     case 'X':
 99       base = base ? '0x': '';
100       value = base + value.toString(16).toUpperCase();
101       break;
102     case 'e':
103       value = value.toExponential().toLowerCase();
104       break;
105     case 'E':
106       value = value.toExponential().toUpperCase();
107       break;
108     case 'f':
109       // Follow Python's example (using 6 as the default)
110       value = value.toPrecision(precision || 7).toLowerCase();
111       break;
112     case 'F':
113       // Follow Python's example (using 6 as the default)
114       value = value.toPrecision(precision || 7).toUpperCase();
115       break;
116     case 'G':
117       value = String(value).toUpperCase();
118       break;
119     case '%':
120       value = (value.toPrecision(7) * 100) + '%';
121       break;
122     case 'n':
123       value = value.toLocaleString();
124       break;
125     case 'g':
126     case 'd':
127     case '':
128     case void 0:
129       value = String(value).toLowerCase();
130       break;
131     default:
132       throw new Error('Unrecognized format type: "{0}"'.fmt(type));
133     }
134 
135     if (align !== '=') {
136       value = sign + value;      
137     }
138 
139     spec = (fill || '') + (align || '') + (minWidth || '') + (precision || '') + (type || '');
140     value = String(value).__fmt__(spec);
141 
142     if (align === '=') {
143       value = sign + value;
144     }
145 
146     return value;
147   }
148
