 39   /** @ignore
 40     The abstract operation Str(key, holder) has access to ReplacerFunction
 41     from the invocation of the stringify method.
 42    */
 43   function Str(key, holder) {
 44     var value;
 45 
 46     // 1. Let value be the result of calling the [[Get]] internal method of
 47     //    holder with argument key.
 48     value = holder && (holder.get && holder.get(key) || holder[key]);
 49 
 50     // 2. If Type(value) is Object, then
 51     if (typeof value === "object") {
 52       // a. Let toJSON be the result of calling the [[Call]] internal method
 53       //    of value with argument toJSON
 54       // b. If IsCallable(toJSON) is true
 55       if (value && Espresso.isCallable(value.toJSON)) {
 56         // i. Let value be the result of calling the [[Call]] internal method
 57         //    of toJSON passing value as the this value and with an argument
 58         //    list consisting of key and value.
 59         value = value.toJSON(key);
 60       }
 61     }
 62 
 63     // 3. If ReplacerFunction is not undefined, then
 64     if (typeof ReplacerFunction !== "undefined") {
 65       // a. Let value be the result of calling the [[Call]] internal method
 66       //    of ReplacerFunction passing holder as the this value and with
 67       //    an argument list containing key and value.
 68       ReplacerFunction.call(holder, key, value);
 69     }
 70 
 71     // 5. If value is null then return "null".
 72     switch (typeof value) {
 73     case 'boolean':
 74     case 'null':
 75       return String(value);
 76     case 'string':
 77       return Quote(value);
 78     case 'number':
 79       return isFinite(value) ? String(value): "null";
 80     case 'object':
 81       if (!value) {
 82         return 'null';
 83       }
 84 
 85       if (Array.isArray(value)) {
 86         return JA(value);
 87       } else {
 88         return JO(value);
 89       }
 90       break;
 91     default:
 92       return void 0;
 93     }
 94   };

 96   /** @ignore
 97     The abstract operation Quote(value) wraps a String value
 98     in double quotes and escapes characters within it.
 99    */
100   function Quote(value) {
101     escapable.lastIndex = 0;
102     return '"' + (escapable.test(value) ?
103       value.replace(escapable, function (a) {
104         var c = abbrev[a];
105         return typeof c === 'string' ?
106           c: '\\u' + ('0000' + a.charCodeAt(0).toString(16)).slice(-4);
107       }): value) + '"';
108   };

110   /** @ignore */
111   function JO(value) {
112     var stepback, K, partial;
113 
114     // 1. If stack contains value then throw a TypeError exception
115     //    because the structure is cyclical.
116     if (stack.indexOf(value) !== -1) {
117       throw new TypeError("Cannot stringify a cyclical structure to JSON.");
118     }
119 
120     // 2. Append value to stack.
121     stack[stack.length] = value;
122 
123     // 3. Let stepback be indent.
124     stepback = indent;
125 
126     // 4. Let indent be the concatenation of indent and gap.
127     indent += gap;
128 
129     // 5. If PropertyList is not undefined, then
130     if (typeof PropertyList !== "undefined") {
131       // a. Let K be PropertyList.
132       K = PropertyList;
133 
134     // 6. Else
135     } else {
136       // a. Let K be an internal List of Strings consisting
137       //    of the names of all the own properties of value whose
138       //    [[Enumerable]] attribute is true.
139       //    The ordering of the Strings should be the same as that used by the
140       //    Object.keys standard built-in function.
141       K = Object.keys(value);
142     }
143 
144     // 7. Let partial be an empty List.
145     partial = [];
146 
147     // 8. For each element P of K.
148     K.forEach(function (P) {
149       var strP = Str(P, value), member;
150       if (typeof strP !== "undefined") {
151         member = Quote(P);
152         member += ":";
153         if (gap !== '') {
154           member += space;
155         }
156         member += strP;
157         partial[partial.length] = member;
158       }
159     });
160 
161     var result;
162     if (partial.length === 0) {
163       result = "{}";
164     } else {
165       if (gap === '') {
166         result = "{" + partial.join(',') + "}";
167       } else {
168         result = '{\n' +
169           indent + partial.join(',\n' + indent) + '\n' +
170          stepback + '}';
171       }
172     }
173 
174     stack.pop();
175     indent = stepback;
176     return result;
177   };

179   /** @ignore */
180   function JA(value) {
181     var stepback, partial, len, index, strP, result;
182 
183     // 1. If stack contains value then throw a TypeError exception
184     //    because the structure is cyclical.
185     if (stack.indexOf(value) !== -1) {
186       throw new TypeError("Cannot stringify a cyclical structure to JSON.");
187     }
188 
189     // 2. Append value to stack.
190     stack[stack.length] = value;
191 
192     // 3. Let stepback be indent.
193     stepback = indent;
194 
195     // 4. Let indent be the concatenation of indent and gap.
196     indent += gap;
197 
198     // 5. Let partial be an empty List.
199     partial = [];
200 
201     // 6. Let len be the result of calling the [[Get]] internal method
202     //    of value with argument "length".
203     len = value.get('length');
204 
205     // 7. Let index be 0.
206     index = 0;
207 
208     // 8. Repeat while index < len
209     while (index < len) {
210       strP = Str(String(index), value);
211       if (typeof strP === "undefined") {
212         partial[partial.length] = "null";
213       } else {
214         partial[partial.length] = strP;
215       }
216       index += 1;
217     }
218 
219     // 9. If partial is empty, then
220     if (partial.length === 0) {
221       result = "[]";
222 
223     // 10. Else
224     } else {
225       if (gap === '') {
226         result = "[" + partial.join(',') + "]";
227       } else {
228         result = '[\n' +
229           indent + partial.join(',\n' + indent) + '\n' +
230          stepback + ']';
231       }
232     }
233 
234     // 11. Remove the last element of stack.
235     stack.pop();
236 
237     // 12. Let indent be stepback.
238     indent = stepback;
239 
240     // 13. Return final.
241     return result;
242   };

244   /** @function
245     @desc
246 
247     The `stringify` function returns a String in JSON
248     format representing an ECMAScript value.
249 
250     @param {Object} value An ECMAScript value.
251     @param {Function|Array} [replacer] Either a function that
252       alters the way objects and arrays are stringified or an
253       array of strings and numbers that acts as a white list
254       for selecteing the object properties that will be stringified.
255     @param {String|Number} [space] Allows the result to have
256       white space injected into it to improve human readability.
257 
258     @returns {String} The ECMAScript value as a JSON string.
259    */
260   function stringify(value, replacer, sp) {
261     var k, v, len, item;
262 
263     // 1. Let stack be an empty List.
264     stack = [];
265 
266     // 2. Let indent be the empty String.
267     indent = '';
268 
269     // 3. Let PropertyList and ReplacerFunction be undefined
270     PropertyList = ReplacerFunction = undefined;
271 
272     // 4. If Type(replacer) is Object, then
273     if (typeof replacer === "object") {
274       // a. If IsCallable(replacer) is true, then
275       if (isCallable(replacer)) {
276         //  i. Let ReplacerFunction be replacer.
277         ReplacerFunction = replacer;
278 
279         // b. Else if the [[Class]] internal property of replacer is "Array", then
280       } else if (Array.isArray()) {
281         //  i. Let PropertyList be an empty internal List
282         PropertyList = [];
283 
284         // ii. For each value v of a property of replacer that has an array
285         //     index property name. The properties are enumerated in the ascending
286         //     array index order of their names.
287         len = replacer.length;
288         for (k = 0; k < len; k += 1) {
289           v = replacer[k];
290           item = undefined;
291           if (typeof v === "string") {
292             item = v;
293           } else if (typeof v === "number") {
294             item = v.toString();
295           } else if (typeof v === "object" &&
296                      (/string/i.test(Object.prototype.toString.call(v)) ||
297                       /number/i.test(Object.prototype.toString.call(v)))) {
298             item = v.toString();
299           }
300           if (typeof item !== "undefined" && PropertyList.indexOf(item) === -1) {
301             PropertyList[PropertyList.length] = item;
302           }
303         }
304       }
305     }
306     // 5. If Type(space) is Object then,
307     if (typeof sp === "object") {
308       // a. If the [[Class]] property of space is "Number" then,
309       if (/number/i.test(Object.prototype.toString.call(sp))) {
310         // i. Let space be ToNumber(space).
311         sp = Number(sp);
312 
313         // b. Else if the [[Class]] internal property of space is "String" then,
314       } else if (/string/i.test(Object.prototype.toString.call(sp))) {
315         // i. Let space be ToString(space)
316         sp = sp.toString();
317       }
318     }
319 
320     // 6. If Type(space) is Number
321     if (typeof sp === "number") {
322       // a. Let space be min(10, ToInteger(space)).
323       sp = Math.min(10, sp);
324       // b. Set gap to a String containing `space` space characters.
325       //    This will be the empty String if space is less than 1.
326       gap = sp < 1 ? '': ' '.repeat(sp);
327 
328     // 7. Else if Type(space) is String
329     } else if (typeof sp === "string") {
330       // a. If the number of characters in space is 10 or less,
331       //    set gap to space otherwise set gap to a String consisting
332       //    of the first 10 characters of space.
333       gap = (sp.length <= 10) ? sp: sp.slice(0, 10);
334 
335     // 8. Else
336     } else {
337       // a. Set gap to the empty String.
338       gap = '';
339     }
340     space = sp;
341 
342     return Str('', {'': value});
343   }

345   /** @ignore */
346   function evaluate(text) {
347     var at = 0,     // The index of the current character
348         ch = ' ',   // The current character
349         escapee = { '"':  '"',
350                     '\\': '\\',
351                     '/':  '/',
352                     b:    '\b',
353                     f:    '\f',
354                     n:    '\n',
355                     r:    '\r',
356                     t:    '\t' },
357     /** @ignore */
358     next = function (c) {
359       // If a c parameter is provided, verify that it matches the current character.
360       if (c && c !== ch) {
361         throw new SyntaxError("Expected '{}' instead of '{}'".fmt(c, ch));
362       }
363 
364       // Get the next character. When there are no more characters,
365       // return the empty string.
366       ch = text.charAt(at);
367       at += 1;
368       return ch;
369     },
370 
371     // Parse a number value.
372     /** @ignore */
373     number = function () {
374       var number,
375           string = '';
376 
377       if (ch === '-') {
378         string = '-';
379         next('-');
380       }
381       while (ch >= '0' && ch <= '9') {
382         string += ch;
383         next();
384       }
385       if (ch === '.') {
386         string += '.';
387         while (next() && ch >= '0' && ch <= '9') {
388           string += ch;
389         }
390       }
391       if (ch === 'e' || ch === 'E') {
392         string += ch;
393         next();
394         if (ch === '-' || ch === '+') {
395           string += ch;
396           next();
397         }
398         while (ch >= '0' && ch <= '9') {
399           string += ch;
400           next();
401         }
402       }
403       number = +string;
404       if (isNaN(number)) {
405         throw new SyntaxError("'{}' is not a number.".fmt(string));
406       } else {
407         return number;
408       }
409     },
410 
411     // Parse a string value.
412     /** @ignore */
413     string = function () {
414       var hex, i, string = '', uffff;
415 
416       // When parsing for string values, we must look for " and \ characters.
417       if (ch === '"') {
418         while (next()) {
419           if (ch === '"') {
420             next();
421             return string;
422           } else if (ch === '\\') {
423             next();
424             if (ch === 'u') {
425               uffff = 0;
426               for (i = 0; i < 4; i += 1) {
427                 hex = parseInt(next(), 16);
428                 if (!isFinite(hex)) {
429                   break;
430                 }
431                 uffff = uffff * 16 + hex;
432               }
433               string += String.fromCharCode(uffff);
434             } else if (typeof escapee[ch] === 'string') {
435               string += escapee[ch];
436             } else {
437               break;
438             }
439           } else {
440             string += ch;
441           }
442         }
443       }
444       throw new SyntaxError("Bad string.");
445     },
446 
447     // Skip whitespace.
448     /** @ignore */
449     white = function () {
450       while (ch && ch <= ' ') {
451         next();
452       }
453     },
454 
455     // true, false, or null.
456     /** @ignore */
457     word = function () {
458       switch (ch) {
459       case 't':
460         next('t');
461         next('r');
462         next('u');
463         next('e');
464         return true;
465       case 'f':
466         next('f');
467         next('a');
468         next('l');
469         next('s');
470         next('e');
471         return false;
472       case 'n':
473         next('n');
474         next('u');
475         next('l');
476         next('l');
477         return null;
478       }
479       throw new SyntaxError("Unexpected character '{}'".fmt(ch));
480     },
481 
482     value,  // Place holder for the value function.
483 
484     // Parse an array value.
485     /** @ignore */
486     array = function () {
487       var array = [];
488 
489       if (ch === '[') {
490         next('[');
491         white();
492         if (ch === ']') {
493           next(']');
494           return array;   // empty array
495         }
496         while (ch) {
497           array.push(value());
498           white();
499           if (ch === ']') {
500             next(']');
501             return array;
502           }
503           next(',');
504           white();
505         }
506       }
507       throw new SyntaxError("Bad Array");
508     },
509 
510     // Parse an object value.
511     /** @ignore */
512     object = function () {
513       var key,
514           object = {};
515 
516       if (ch === '{') {
517         next('{');
518         white();
519         if (ch === '}') {
520           next('}');
521           return object;   // empty object
522         }
523         while (ch) {
524           key = string();
525           white();
526           next(':');
527           if (Object.hasOwnProperty.call(object, key)) {
528             throw new SyntaxError('Duplicate key "{}"'.fmt(key));
529           }
530           object[key] = value();
531           white();
532           if (ch === '}') {
533             next('}');
534             return object;
535           }
536           next(',');
537           white();
538         }
539       }
540       throw new SyntaxError("Bad Object");
541     };
542 
543     // Parse a JSON value. It could be an object, an array, a string, a number,
544     // or a word.
545     /** @ignore */
546     value = function () {
547       white();
548       switch (ch) {
549       case '{':
550         return object();
551       case '[':
552         return array();
553       case '"':
554         return string();
555       case '-':
556         return number();
557       default:
558         return ch >= '0' && ch <= '9' ? number() : word();
559       }
560     };
561 
562     var ret = value();
563     white();
564     if (ch) {
565       throw new SyntaxError("Unexpected character '{}'".fmt(ch));
566     }
567     return ret;
568   }

570   /** @function
571     @desc
572 
573     The `parse` function parses a JSON text and produces an
574     ECMAScript value.
575 
576     @param {String} text The JSON text to parse.
577     @param {Function} [reviver] A function that takes two
578       values (key and value). It can filter and transform the
579       results. It is called with each of the key/value pairs
580       produced by the parse, and its return value is used
581       instead of the original value. If it returns what it
582       recieved, the structure is not modified. If it returns
583       `undefined`, then the property is deleted from the result.
584 
585     @returns {Object} The JSON text as an ECMAScript value.
586    */
587   function parse(text, reviver) {
588     var o = evaluate(text);
589 
590     /** @ignore */
591     var Walk = function (holder, key) {
592       var val = holder[key], k, v;
593 
594       if (Espresso.hasValue(val)) {
595         for (k in val) {
596           if (val.hasOwnProperty(k)) {
597             v = Walk(val, k);
598             if (typeof v === "undefined") {
599               delete val[k];
600             } else {
601               val[k] = v;
602             }
603           }
604         }
605       }
606       return reviver.call(holder, key, val);
607     };
608 
609     return Espresso.isCallable(reviver) ?
610       Walk({ '': o }, ''): o;
611   }
