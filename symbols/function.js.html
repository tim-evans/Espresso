  5   /**
  6     If the attribute being mixed in exists on the
  7     Object being mixed in, the function marked as
  8     inferior will **not** be mixed in.
  9 
 10     Also, if the base function is inferior, it
 11     will be overriden.
 12 
 13     @param {Function} [condition] If it returns `true`,
 14       the function is inferior. Otherwise, it isn't.
 15     @returns {Function} The reciever.
 16    */
 17   inferior: function (condition) {
 18     this._ = this._ || {};
 19     this.isInferior = arguments.length === 1 ?
 20       (Espresso.isCallable(condition) ? condition() : condition) : true;
 21     if (!this.isInferior) { return this; }
 22 
 23     /** @ignore */
 24     this._.inferior = function (template, value, key) {
 25       return (!template[key] || template[key].isInferior) ? value: template[key];
 26     };
 27 
 28     return this;
 29   },
 30 
 31   /**
 32     Provides a mechanism to alias a function with
 33     other names on the object.
 34 
 35     Any arguments passed in will be used as aliases
 36     for the function. Each of these aliases will be
 37     references to the original, meaning that all of
 38     them will be indistinguishable and if one is
 39     altered in place, then all will be.
 40 
 41     @param {...} aliases The aliases this function has.
 42     @returns {Function} The reciever.
 43    */
 44   alias: function () {
 45     this._ = this._ || {};
 46 
 47     var aliases = Espresso.toArray(arguments),
 48         idx = aliases.length, mixin;
 49 
 50     /** @ignore */
 51     this._.alias = function (template, value, key) {
 52       delete value._.alias; // Remove this to prevent recursion.
 53       while (idx--) {
 54         mixin = {};
 55         mixin[aliases[idx]] = value;
 56         mix(mixin).into(template);
 57       }
 58       return value;
 59     };
 60 
 61     return this;
 62   },
 63 
 64   /**
 65     Refine provides `super` functionality to a function.
 66     When the decorated function is called, it will have it's
 67     first argument bound to the function this one will override.
 68 
 69     If this function will not override anything, then the first
 70     argument will be an empty function that returns nothing.
 71 
 72     The `super` function will always be in the current scope of
 73     the function being called at the moment. Since the scope is
 74     maintained for you, you **must** make sure that scope that the
 75     first in the chain is what you want all down the chain. This
 76     is the typical behaviour of `super` in other languages;
 77     therefore it is done for you, and forces you into that
 78     situation.
 79 
 80     NOTE: If you try to rebind the property using
 81           {@link Function#bind}, it will _not_ work.
 82 
 83     @returns {Function} The reciever.
 84    */
 85   refine: function () {
 86     this._ = this._ || {};
 87 
 88     var empty = function () {};
 89 
 90     /** @ignore */
 91     this._.refine = function (template, value, key) {
 92       var base = template[key] || empty;
 93       if (!Espresso.isCallable(base)) {
 94         return value;
 95       }
 96 
 97       /** @ignore */
 98       var lambda = function () {
 99         return value.apply(this, [base.bind(this)].concat(Espresso.toArray(arguments)));
100       };
101 
102       // Copy over properties on `value`
103       for (var k in value) {
104         if (value.hasOwnProperty(k)) {
105           lambda[k] = value[k];
106         }
107       }
108       return lambda;
109     };
110     return this;
111   },

113   /**
114     Marks the function as a computed property.
115     You may now use the function for `get` and `set`.
116 
117     @param {...} dependentKeys The property paths to be
118       notified when anything gets published to them.
119     @returns {Function} The reciever.
120    */
121   property: function () {
122     this._ = this._ || {};
123 
124     this.isProperty = true;
125     this.dependentKeys = Array.prototype.slice.apply(arguments);
126 
127     /** @ignore */
128     this._.property = function (template, value, key) {
129       var i = 0, len = value.dependentKeys.length, object, property, iProperty,
130           /** @ignore */
131           notifier = function () {
132             this.set(key);
133           };
134 
135       for (i = 0; i < len; i += 1) {
136         property = value.dependentKeys[i];
137         object = template;
138 
139         if (property.indexOf('.') !== -1) {
140           iProperty = property.lastIndexOf('.');
141           object = Espresso.getObjectFor(property.slice(0, iProperty));
142           property = property.slice(iProperty + 1);
143         }
144 
145         if (object && object.subscribe && object.publish) {
146           object.subscribe(property, notifier, { synchronous: true });
147         }
148       }
149 
150       return value;
151     };
152 
153     return this;
154   },

191   /**
192     Bind the value of `this` on a function before hand,
193     with any extra arguments being passed in as initial
194     arguments.
195 
196     This implementation conforms to the ECMAScript 5
197     standard.
198 
199         var Person = mix({
200           name: 'nil',
201           greet: function (greeting) {
202             alert(greeting.fmt(this.name));
203           }
204         }).into({});
205 
206         var wash = mix(Person).into({
207           name: 'Hoban Washburne'
208         });
209 
210         var mal = mix(Person).into({
211           name: 'Malcolm Reynolds'
212         });
213 
214         mal.greet("Hello, {}!");
215         // -> "Hello, Malcolm Reynolds!"
216 
217         var greet = mal.greet.bind(wash);
218         greet("Howdy, {}!");
219         // -> "Howdy, Hoban Washburne!"
220 
221     @param {Object} thisArg The value to bind `this` to on the function.
222     @returns {Function} The function passed in, wrapped to ensure `this`
223       is the correct scope.
224    */
225   bind: function (self) {
226     var Target, A;
227 
228     // 1. Let Target be the this value.
229     Target = this;
230 
231     // 2. If IsCallable(Target) is false, throw a TypeError exception
232     if (!Espresso.isCallable(Target)) {
233       throw new TypeError("The Target is not callable.");
234     }
235 
236     // 3. Let A be a new (possibly empty) internal list of
237     //    all argument values provided after self
238     //    (arg1, arg2, etc), in order
239     A = Espresso.toArray(arguments).slice(1);
240 
241     var bound = function () {
242       
243       if (this instanceof bound) {
244         // 15.3.4.5.2 [[Construct]]
245         // When the [[Construct]] internal method of a function object, F,
246         // that was created using the bind function is called with a list of
247         // arguments ExtraArgs, the following steps are taken:
248         var Type = function () {}, that;
249         Type.prototype = Target.prototype;
250         that = new Type();
251 
252         Target.apply(self, A.concat(Espresso.toArray(arguments)));
253         return that;
254       } else {
255         // 15.3.4.5.1 [[Call]]
256         // When the [[Call]] internal method of a function object, F,
257         // which was created using the bind function is called with a this
258         // value and a list of arguments ExtraArgs, the following steps are taken:
259         // 1. Let boundArgs be the value of F's [[BoundArgs]] internal property.
260         // 2. Let boundThis be the value of F's [[BoundThis]] internal property.
261         // 3. Let target be the value of F's [[TargetFunction]] internal property.
262         return Target.apply(self, A.concat(Espresso.toArray(arguments)));
263       }
264     };
265     return bound;
266   }.inferior(false), // bind seems to be broken on all browsers

268   /**
269     Currying transforms a function by transforming by composing it into
270     functions that each take a subset of the arguments of the whole.
271     The aggregate of all of the arguments passed into curried function
272     and the actual function call is what will be provided to the function
273     that's being curried.
274 
275     This effectively means that you can transform a very simple call into
276     atomic calls as follows:
277 
278         var mult = function () {
279           return Espresso.toArray(arguments).reduce(function (E, x) { return E * x; }, 1);
280         };
281 
282         alert(mult.curry(2, 2)());
283         // => 4
284 
285     Or specify a function that is a subset of the first:
286 
287         var add = function () {
288           return Espresso.toArray(arguments).reduce(function (E, x) { return E + x; }, 0);
289         };
290 
291         var inc = add.curry(1);
292 
293         alert(inc(5));
294         // => 6
295 
296     In layman's terms, `curry` will prepopulate arguments for a function.
297     @param {...} args Arguments that will be curried to the function.
298     @returns {Function} The function with the arguments provided saved for later.
299    */
300   curry: function () {
301     var Target, A;
302 
303     // 1. Let Target be the this value.
304     Target = this;
305 
306     // 2. If IsCallable(Target) is false, throw a TypeError exception
307     if (!Espresso.isCallable(Target)) {
308       throw new TypeError("The Target is not callable.");
309     }
310 
311     // 3. Let A be a new (possibly empty) internal list of
312     //    all argument values (arg1, arg2, etc), in order
313     A = Espresso.toArray(arguments);
314 
315     return function () {
316       return Target.apply(this, A.concat(Espresso.toArray(arguments)));
317     };
318   }
