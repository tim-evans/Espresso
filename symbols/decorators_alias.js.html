1 mix(/** @scope Espresso */{
  2 
  3   /**
  4     Provides a mechanism to alias an object with
  5     using other names.
  6 
  7     Any arguments passed in after the target will
  8     be used as aliases for the target. Each of these
  9     aliases will be references to the original, meaning
 10     that all of them will be indistinguishable and if
 11     one of them is altered in place, then all will be.
 12 
 13     @param {Object} target The target to apply this decorator to.
 14     @param {...} aliases The aliases this object has.
 15     @returns {Object} The reciever.
 16    */
 17   alias: function (target) {
 18     target._ = target._ || {};
 19 
 20     var aliases = Espresso.A(arguments).slice(1),
 21         idx = aliases.length, mixin;
 22 
 23     /** @ignore */
 24     target._.alias = function (template, value, key) {
 25       delete value._.alias; // Remove this to prevent recursion.
 26       while (idx--) {
 27         mixin = {};
 28         mixin[aliases[idx]] = value;
 29         mix(mixin).into(template);
 30       }
 31       return value;
 32     };
 33 
 34     return target;
 35   }
 36 
 37 }).into(Espresso);
