  3 /** @namespace
  4   Implements the Observer / Publish-Subscribe pattern.
  5 
  6   Subscribe to events that are published to objects that
  7   mixin this, and you'll be notified when the events come
  8   in. If something is published and there are no handlers
  9   for that specific event, there is a `unpublishedEvent`
 10   function that will be called whenever an event doesn't
 11   have any subscribers.
 12 
 13   Publishing an event will use the first argument as the
 14   event to trigger, and call all the subscription handlers
 15   with all of the arguments passed into that `publish`.
 16 
 17   Subscribing to an event requires the event that it would
 18   like to recieve events from and the callback at minimum.
 19 
 20   If extra configuration is wanted, the `options` hash
 21   provides a way to dynamically have events delivered or
 22   ignored beforehand (possibly providing lint-checking before
 23   the event is delivered), and whether the event should
 24   be delivered synchronously or asynchronously. (By default,
 25   it's asynchronous).
 26 
 27   @example
 28       var Clock = mix(Espresso.Subscribable, {
 29         tick: function () {
 30           this.time = Date.now();
 31         }
 32       }).into({});
 33 
 34       Clock.subscribe("tick", Clock.tick);
 35       setInterval(Clock.publish.bind(Clock, "tick"), 1000);
 36 
 37  */
 38 Espresso.Subscribable = /** @lends Espresso.Subscribable# */{
 39 
 40   /**
 41     Walk like a duck.
 42     @type Boolean
 43    */
 44   isSubscribable: true,
 45 
 46   /** @private */
 47   __subscriptions__: null,
 48 
 49   /**
 50     Subscribe to an event.
 51 
 52     @param {Object} event The event to subscribe to.
 53     @param {Function} handler The handler to call when the event is published.
 54     @param {Object} [options] Optional parameters.
 55       @param {Boolean} [options.synchronous] Whether the handler should be called synchronously or not. Defaults to asynchronous calls.
 56       @param {Function} [options.condition] A mechanism to refine whether a specific event is wanted. Return true if you would like the event, and false if you don't.
 57     @returns {Object} The reciever.
 58    */
 59   subscribe: function (event, handler, options) {
 60     if (!Espresso.isCallable(handler)) {
 61       throw new TypeError("{} is not callable.".format(handler));
 62     }
 63 
 64     var subscriptions = this.__subscriptions__ || {};
 65     if (!subscriptions[event]) {
 66       subscriptions[event] = [];
 67     }
 68 
 69     if (options && options.condition && !Espresso.isCallable(options.condition)) {
 70       delete options.condition;
 71     }
 72 
 73     options = mix({
 74       condition: Espresso.inferior(function () { return true; })
 75     }).into(options || {});
 76 
 77     subscriptions[event].push(mix(options, {
 78       subscriber: handler
 79     }).into({}));
 80 
 81     this.__subscriptions__ = subscriptions;
 82     return this;
 83   },
 84 
 85   /**
 86     Unsubscribe from an event.
 87 
 88     @param {Object} event The event to subscribe to.
 89     @param {Function} handler The handler to call when the event is published.
 90     @returns {Object} The reciever.
 91    */
 92   unsubscribe: function (event, handler) {
 93     var subscriptions = this.__subscriptions__, handlers, i, len;
 94     if (subscriptions && subscriptions[event]) {
 95       handlers = subscriptions[event];
 96       for (i = 0, len = handlers.length; i < len; i += 1) {
 97         if (handlers[i].subscriber === handler) {
 98           subscriptions[event].splice(i, 1);
 99           break;
100         }
101       }
102     }
103     return this;
104   },
105 
106   /**
107     Gets called when an event has no subscribers to it.
108 
109     Override to handle the case when nothing is published.
110     (There are no subscribers for an event.)
111 
112     Any parameters passed to the event are also passed into
113     the function. All unpublished events are invoked immediately
114     rather than `defer`red.
115 
116     @param {Object} event The event that was ignored.
117     @returns {void}
118    */
119   unpublishedEvent: function (event) {},
120 
121   /**
122     Publish an event, passing all arguments along to the subscribed functions.
123 
124     @param {Object} event The event to publish.
125     @returns {Object} The reciever.
126    */
127   publish: function (event) {
128     var subscriptions = this.__subscriptions__,
129         args = arguments, subscriber, published = false;
130     if (subscriptions && subscriptions[event]) {
131       subscriptions[event].forEach(function (subscription) {
132         if (subscription.condition.apply(this, args)) {
133           subscriber = subscription.subscriber;
134           if (subscription.synchronous) {
135             subscriber.apply(this, args);
136           } else {
137             Espresso.defer(subscriber, args, this);
138           }
139           published = true;
140         }
141       }, this);
142     }
143     if (!published && Espresso.isCallable(this.unpublishedEvent)) {
144       this.unpublishedEvent.apply(this, arguments);
145     }
146     return this;
147   }
148 };
