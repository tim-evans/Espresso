 10   /**
 11     Iterates over every character in a string.
 12     Required by {@link Espresso.Enumerable}.
 13 
 14     @param {Function} callback The callback to call for each element.
 15     @param {Object} that The Object to use as this when executing the callback.
 16 
 17     @returns {void}
 18     @example
 19       "boom".forEach(alert);
 20       // => 'b'
 21       // => 'o'
 22       // => 'o'
 23       // => 'm'
 24    */
 25   forEach: function (lambda, that) {
 26     var i = 0, len = this.length;
 27 
 28     if (!Espresso.isCallable(lambda)) {
 29       throw new TypeError("{} is not callable.".fmt(lambda));
 30     }
 31     for (; i < len; i += 1) {
 32       lambda.call(that, this.charAt(i), i, this);
 33     }
 34   },

 42     @returns {String} The character at index idx.
 43    */

 48   /**
 49     Capitalize a string.
 50 
 51     @returns {String} The string, capitalized.
 52     @example
 53       ['toast', 'cheese', 'wine'].forEach(function (food) {
 54         alert(food.capitalize());
 55       });
 56       // => "Toast"
 57       // => "Cheese"
 58       // => "Wine"

 60   capitalize: function () {
 61     return this.get(0).toUpperCase() + this.slice(1);
 62   },

 64   /**
 65     Returns the string repeated the specified number of times.
 66 
 67     @param {Number} n The number of times to repeat this string.
 68     @param {String} [separator] The separator to put between each iteration of the string.
 69     @returns {String} The string repeated n times.
 70     @example
 71       alert("bacon".times(5));
 72       // => "baconbaconbaconbaconbacon"
 73 
 74     @example
 75       alert("crunchy".times(2, " bacon is "));
 76       // => "crunchy bacon is crunchy"
 77    */
 78   times: function (n, sep) {
 79     sep = sep || '';
 80     return n < 1 ? '': (new Array(n)).join(this + sep) + this;
 81   },

 97   /** @function
 98 
 99     Unescapes any escaped HTML strings into their readable
100     forms.
101 
102     @returns {String} The unescaped string.
103    */
104   unescapeHTML: (function () {
105     // The entity table. It maps entity names to characters.
106     var entity = {
107       quot: '"',
108       lt:   '<',
109       gt:   '>',
110       amp:  '&',
111       apos: "'"
112     }, re = /&([^&;]+);/g;
113 
114     // Replaces entity characters with their
115     // more commonplace cousins:
116     //  eg. " => "
117     return function () {
118       return this.replace(re,
119         function (a, b) {
120           var r = entity[b];
121           return typeof r === 'string' ? r : a;
122         }
123       );
124     };
125   }()).inferior(),

164   /**
165     Format formats a string in the vein of Python's format,
166     Ruby #{templates}, and .NET String.Format.
167 
168     To write { or } in your Strings, just double them, and
169     you'll end up with a single one.
170 
171     If you have more than one argument, then you can reference
172     by the argument number (which is optional on a single argument).
173 
174     If you want to tie into this, and want to specify your own
175     format specifier, override __fmt__ on your object, and it will
176     pass you in the specifier (after the colon). You return the
177     string it should look like, and that's it!
178 
179     For an example of an formatting extension, look at the Date mix.
180     It implements the Ruby/Python formatting specification for Dates.
181 
182     @returns {String} The formatted string.
183     @example
184       alert("b{0}{0}a".fmt('an'));
185       // => "banana"
186 
187     @example
188       var kitty = Espresso.Template.extend({
189         name: "Mister Mittens",
190         weapons: ["lazzors", "shuriken", "rainbows"],
191 
192         fight: function (whom) {
193           return "fightin' the {} with his {}.".fmt(
194             whom, this.weapons[Math.floor(Math.random() * this.weapons.length)]);
195         }
196       });
197 
198       alert("{0.name} is {1}".fmt(kitty, kitty.fight('zombies')));
199       // => "Mister Mittens is fightin' the zombies with ..."
200 
201     @example
202       alert("I love {pi:.2}".fmt({ pi: 22 / 7 }));
203       // => "I love 3.14"
204 
205     @example
206       alert("The {confection.type} is {confection.descriptor}.".fmt({
207         confection: {
208           type: 'cake',
209           descriptor: 'a lie'
210         }
211       }));
212       // => "The cake is a lie."
213 
214     @example
215       alert(":-{{".fmt());  // Double {{ or }} to escape it.
216       // => ":-{"
217    */
218   fmt: function () {
219     var args = Array.from(arguments);
220     args.unshift(this.toString());
221     return Espresso.Formatter.fmt.apply(Espresso.Formatter, args);
222   },
223 
224   /**
225     Format a string according to a format specifier.
226     This is a function called by Formatter, 
227     A valid specifier can have:
228     [[fill]align][minimumwidth]
229 
230     @param {String} spec The specifier string.
231     @returns {String} The string formatted using the format specifier.
232    */
233   __fmt__: function (spec) {
234     var match = spec.match(Espresso.Formatter.SPECIFIER),
235         align = match[1],
236         fill = match[2] || ' ',
237         minWidth = match[6] || 0,
238         maxWidth = match[7] || null, len, before, after, value;
239 
240     if (align) {
241       align = align.slice(-1);
242     }
243 
244     len = Math.max(minWidth, this.length);
245     before = len - this.length;
246     after = 0;
247 
248     switch (align) {
249     case '<':
250       after = before;
251       before = 0;
252       break;
253     case '^':
254       after = Math.ceil(before / 2);
255       before = Math.floor(before / 2);
256       break;
257     }
258 
259     value = this;
260     if (Espresso.hasValue(maxWidth)) {
261       maxWidth = +maxWidth.slice(1);
262       if (!isNaN(maxWidth)) {
263         value = value.slice(0, maxWidth);
264       }
265     }
266 
267     return fill.times(before) + value + fill.times(after);
268   },
