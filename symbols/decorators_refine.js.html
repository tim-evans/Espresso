1 mix(/** @scope Espresso */{
  2 
  3   /**
  4     Refine allows for function-by-function refinements that
  5     reopens the function implementation without editing the
  6     original function's contents. With this, you can implement
  7     OO constructs like abstract base classes.
  8 
  9     Refinements to a function recieve a prepended argument to
 10     the argument list which is the original function that
 11     is being refined (if there isn't an original function that's
 12     being refined, a empty function will be provided for consistency).
 13 
 14     Calling the refined function should be done like so:
 15 
 16         Machiatto = mix({
 17           pull: function (original) {
 18             var espresso = original();
 19             return espresso + milk;
 20           }.refine()
 21         }).into(Espresso);
 22 
 23     Provide arguments as-is, omit arguments, or add arguments
 24     to the function. It'll be just like it's being called normally.
 25 
 26     NOTE: If you try to rebind the property using
 27           {@link Function#bind}, it will _not_ work.
 28 
 29     @param {Function} target The target to apply this decorator to.
 30     @returns {Function} The reciever.
 31    */
 32   refine: function (target) {
 33     if (!Espresso.isCallable(target)) return target;
 34 
 35     target._ = target._ || {};
 36 
 37     var empty = function () {};
 38 
 39     /** @ignore */
 40     target._.refine = function (template, value, key) {
 41       var base = template[key] || empty;
 42       if (!Espresso.isCallable(base)) {
 43         return value;
 44       }
 45 
 46       /** @ignore */
 47       var lambda = function () {
 48         return value.apply(this, [base.bind(this)].concat(Espresso.A(arguments)));
 49       };
 50 
 51       // Copy over function properties
 52       for (var k in value) {
 53         if (value.hasOwnProperty(k)) {
 54           lambda[k] = value[k];
 55         }
 56       }
 57       return lambda;
 58     };
 59     return target;
 60   }
 61 
 62 }).into(Espresso);
