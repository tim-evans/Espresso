 26   /**
 27     Iterator over the Array.
 28 
 29     Implemented to be in conformance with ECMA-262 Edition 5,
 30     so you will use the native `forEach` where it exists.
 31 
 32     @param {Function} lambda The callback to call for each element.
 33     @param {Object} [self] The Object to use as this when executing the callback.
 34     @returns {void}
 35    */
 36   forEach: function (lambda, self) {
 37     var len, k;
 38 
 39     // 3. Let len be ToUint32(lenValue).
 40     len = this.length;
 41 
 42     // 4. If IsCallable(lambda) is false, throw a TypeError exception
 43     if (!Espresso.isCallable(lambda)) {
 44       throw new TypeError("{} is not callable.".fmt(lambda));
 45     }
 46 
 47     // 6. Let k be 0.
 48     k = 0;
 49 
 50     // 7. Repeat, while k < len
 51     while (k < len) {
 52       // c. If kPresent is true, then
 53       if (this.hasOwnProperty(k)) {
 54         //  i. Let kValue be the result of calling the [[Get]]
 55         //     internal method of O with argument Pk.
 56         // ii. Call the [[Call]] internal method of lambda
 57         //     with T as the this value and argument list
 58         //     containing kValue, k, and O.
 59         lambda.call(self, this[k], k, this);
 60       }
 61 
 62       // d. Increase k by 1.
 63       k += 1;
 64     }
 65 
 66     // 8. Return
 67   }.inferior(),

 69   /**
 70     Shim for Internet Explorer, which provides no `indexOf` for
 71     Array prototypes.
 72 
 73     @param {Object} o The object to test.
 74     @param {Number} [fromIndex] The index to start looking at for the element.
 75     @returns {Number} The first index of an item.
 76    */
 77   indexOf: function (o, fromIndex) {
 78     var i = 0, len = this.length;
 79     fromIndex = fromIndex || 0;
 80     i = fromIndex >= 0 ? fromIndex:
 81                          Math.max(i, len - Math.abs(fromIndex));
 82     for (; i < len; i += 1) {
 83       if (o === this[i]) {
 84         return i;
 85       }
 86     }
 87     return -1;

 90   /**
 91     Reduce the content of an array down to a single
 92     value (starting from the end and working backwards).
 93 
 94     @param {Function} lambda The lambda that performs the reduction.
 95       @param {Object} lambda.value The value of the enumerated item.
 96       @param {Object} lambda.key The key of the enumerated item.
 97       @param {Object} lambda.self The object being enumerated over.
 98     @param {Object} [seed] The seed value to provide for the first time.
 99     @returns {Object} The reduced output.
100    */
101   reduceRight: function (lambda, seed) {
102     var shouldSeed = (arguments.length === 1),
103         len = this.length, v;
104 
105     // 4. If IsCallable(lambda) is false, throw a TypeError exception
106     if (!Espresso.isCallable(lambda)) {
107       throw new TypeError("{} is not callable.".fmt(lambda));
108     }
109 
110     while (len-- > 0) {
111       v = this[len];
112       if (shouldSeed) {
113         seed = v;
114         shouldSeed = false;
115       } else {
116         seed = lambda(seed, v, len, this);
117       }
118     }
119 
120     // 5. If len is 0 and seed is not present, throw a TypeError exception.
121     if (shouldSeed) {
122       throw new TypeError("There was nothing to reduce!");
123     }
124     return seed;

127   /**
128     Shim for Internet Explorer, which provides no reverse for
129     Array prototypes. Note: the Array is reversed in-place.
130 
131     @returns {Array} The array in reverse order.
132 
133     @see ECMA-262 15.4.4.8 Array.prototype.reverse()
134 
135     @example
136       var racecar = "racecar".split('');
137       alert(racecar.reverse().join(''));
138       // => 'racecar'
139    */
140   reverse: function () {
141     var O, len, middle,
142         lower, upper,
143         lowerP, upperP,
144         upperValue, lowerValue;
145 
146     // 1. Let O be the result of calling ToObject
147     //    passing this value as the argument.
148     O = this;
149 
150     // 3. Let len be ToUint(lenVal)
151     len = this.length;
152 
153     // 4. Let middle be floor(len/2)
154     middle = Math.floor(len / 2);
155 
156     // 5. Let lower be 0.
157     lower = 0;
158 
159     // 6. Repeat, while lower !== middle
160     while (lower !== middle) {
161       // a. Let upper be len - lower - 1.
162       upper = len - lower - 1;
163 
164       // b. Let upperP be ToString(upper).
165       upperP = upper.toString();
166 
167       // c. Let lowerP be ToString(lower).
168       lowerP = lower.toString();
169 
170       // d. Let lowerValue be the result of calling the [[Get]]
171       //    intenal method of O with argument lowerP
172       lowerValue = this[lowerP];
173       
174       // e. Let upperValue be the result of calling the [[Get]]
175       //    intenal method of O with argument upperP
176       upperValue = this[upperP];
177 
178       // h. If lowerExists is true and upperExists is true, then
179       //     i. Call the [[Put]] internal method of O with arguments
180       //        lowerP, upperValue, and true.
181       //     i. Call the [[Put]] internal method of O with arguments
182       //        upperP, lowerValue, and true.
183       O[lowerP] = upperValue;
184       O[upperP] = lowerValue;
185 
186       // l. Increase lower by 1.
187       lower += 1;
188     }
189 
190     // 7. Return 0.
191     return O;
192   }.inferior(),

194   /**
195     Shim for the last index that the object is found at.
196     Returns -1 if the item is not found.
197 
198     @param searchElement The item to look for.
199     @param [fromIndex] The index to begin searching from.
200     @returns {Number} The last index of an item.
201 
202     @see ECMA-262 15.4.4.15 Array.prototype.lastIndexOf(searchElement [, fromIndex ])
203    */
204   lastIndexOf: function (searchElement, fromIndex) {
205     var k = 0, len = this.length, n;
206 
207     // 4. If len is 0, return -1.
208     if (len === 0) {
209       return -1;
210     }
211 
212     // 5. If argument fromIndex was passed, let n be
213     //    ToInteger(fromIndex); else let n be len.
214     n = fromIndex || len;
215 
216     // 6. If n >= 0, then let k be min(n, len - 1).
217     if (n > 0) {
218       k = Math.min(n, len - 1);
219 
220     // 7. Else, n < 0
221     } else {
222       // a. Let k be len - abs(n).
223       k = len - Math.abs(n);
224     }
225 
226     // 8. Repeat, while k >= 0
227     while (k >= 0) {
228       // a. Let kPresent be the result of calling the [[HasProperty]]
229       //    internal method of O with argument toString(k).
230       // b. If kPresent is true, then
231         //   i. Let elementK be the result of calling the [[Get]]
232         //      internal method of O with the argument toString(k).
233         //  ii. Let same be the result of applying the
234         //      Strict Equality Comparision Algorithm to
235         //      searchElement and elementK.
236         // iii. If same is true, return k.
237       if (this[k.toString()] === searchElement) {
238         return k;
239       }
240 
241       // c. Decrease k by 1.
242       k -= 1;
243     }
244     return -1;

247   /**
248     Returns a new array that's a one-dimensional flattening of this
249     array (recursively). That is, for every element that's an array,
250     extract its elements into the new array. If the optional level
251     arguments determines the level of recursion to flatten.
252 
253     @param {Number} [level] The maximum level of recursion.
254     @returns {Array} The flattened array.
255     @example
256       var arr = [1, [2, [3, [4, [5]]]]];
257       alert(arr.flatten());
258       // => [1, 2, 3, 4, 5]
259 
260       alert(arr.flatten(2));
261       // => [1, 2, 3, [4, [5]]];
262    */
263   flatten: function (level) {
264     var ret = [], hasLevel = arguments.length !== 0;
265     if (hasLevel && level === 0) {
266       return this;
267     }
268 
269     this.forEach(function (v) {
270       if (Array.isArray(v)) {
271         if (hasLevel) {
272           ret = ret.concat(v.flatten(level - 1));
273         } else {
274           ret = ret.concat(v.flatten());
275         }
276       } else {
277         ret[ret.length] = v;
278       }
279     });
280 
281     return ret;
282   },
