  3 /** @namespace
  4   This mixin defines an enumerable interface that is
  5   based off of the ECMAScript 5 specification.
  6 
  7   If any of the functions on this interface are defined
  8   by the host object, they will *not* be applied, with the
  9   assumption that the host object has a better implementation
 10   and the same characteristics.
 11 
 12   @requires `forEach`- the enumerator over the collection.
 13  */
 14 Espresso.Enumerable = /** @lends Espresso.Enumerable# */{
 15 
 16   /**
 17     Walk like a duck.
 18     @type Boolean
 19    */
 20   isEnumerable: true,
 21 
 22   /** @function
 23     @desc
 24     Iterates over each item on the Enumerable.
 25 
 26     The Function `forEach` should follow the specification as
 27     defined in the ECMAScript 5 standard. All function using
 28     `forEach` in the Enumerable mixin depend on it being this way.
 29 
 30     @param {Function} lambda The callback to call for each element.
 31       @param {Object} lambda.value The value of the enumerated item.
 32       @param {Object} lambda.key The key of the enumerated item.
 33       @param {Object} lambda.self The object being enumerated over.
 34     @param {Object} [self] The Object to use as this when executing the callback.
 35     @returns {void}
 36    */
 37   forEach: Espresso.inferior(function (lambda, that) {
 38     throw new Error("You MUST override Espresso.Enumerable.forEach to be able " +
 39                     "to use the Enumerable mixin.");
 40   }),
 41 
 42   /** @function
 43     @desc
 44     Returns an array where each value on the enumerable
 45     is mutated by the lambda function.
 46 
 47     @param {Function} lambda The lambda that transforms an element in the enumerable.
 48       @param {Object} lambda.value The value of the enumerated item.
 49       @param {Object} lambda.key The key of the enumerated item.
 50       @param {Object} lambda.self The object being enumerated over.
 51     @param {Object} [self] The value of `this` inside the lambda.
 52     @returns {Array} The collection of results from the map function.
 53     @example
 54       var cube = function (n) { return n * n * n };
 55       alert([1, 2, 3, 4].map(cube));
 56       // -> [1, 8, 27, 64]
 57    */
 58   map: Espresso.inferior(function (lambda, self) {
 59     var arr = [];
 60 
 61     // 4. If IsCallable(lambda) is false, throw a TypeError exception
 62     if (!Espresso.isCallable(lambda)) {
 63       throw new TypeError("{} is not callable.".format(lambda));
 64     }
 65 
 66     this.forEach(function (k, v) {
 67       arr.push(lambda.call(self, k, v, this));
 68     }, this);
 69     return arr;
 70   }),
 71 
 72   /** @function
 73     @desc
 74     Reduce the content of an enumerable down to a single value.
 75 
 76     @param {Function} lambda The lambda that performs the reduction.
 77       @param {Object} lambda.value The value of the enumerated item.
 78       @param {Object} lambda.key The key of the enumerated item.
 79       @param {Object} lambda.self The object being enumerated over.
 80     @param {Object} [seed] The seed value to provide for the first time.
 81     @returns {Object} The reduced output.
 82     @example
 83       var range = mix(Espresso.Enumerable, {
 84         begin: 0,
 85         end: 0,
 86 
 87         forEach: function (lambda, self) {
 88           var i = 0;
 89           for (var v = this.begin; v <= this.end; v++) {
 90             lambda.call(self, v, i++, this);
 91           }
 92         },
 93 
 94         create: function (begin, end) {
 95           return mix(this, { begin: begin, end: end }).into({});
 96         }
 97       }).into({});
 98 
 99       var multiply = function (a, b) { return a * b; };
100       var factorial = function (n) {
101         return range.create(1, n).reduce(multiply);
102       }
103 
104       alert("5! is {}".format(factorial(5)));
105       alert("120! is {}".format(factorial(120)));
106    */
107   reduce: Espresso.inferior(function (lambda, seed) {
108     var shouldSeed = (arguments.length === 1),
109         self = this;
110 
111     // 4. If IsCallable(lambda) is false, throw a TypeError exception
112     if (!Espresso.isCallable(lambda)) {
113       throw new TypeError("{} is not callable.".format(lambda));
114     }
115 
116     this.forEach(function (v, k) {
117       if (shouldSeed) {
118         seed = v;
119         shouldSeed = false;
120       } else {
121         seed = lambda(seed, v, k, self);
122       }
123     });
124 
125     // 5. If len is 0 and seed is not present, throw a TypeError exception.
126     if (shouldSeed) {
127       throw new TypeError("There was nothing to reduce!");
128     }
129     return seed;
130   }),
131 
132   /** @function
133     @desc
134     Returns all elements on the Enumerable for which the
135     input function returns true for.
136 
137     @param {Function} lambda The function to filter the Enumerable.
138       @param {Object} lambda.value The value of the enumerated item.
139       @param {Object} lambda.key The key of the enumerated item.
140       @param {Object} lambda.self The object being enumerated over.
141     @param {Object} [self] The value of `this` inside the lambda.
142     @returns {Object[]} An array with the values for which `lambda` returns `true`
143    */
144   filter: Espresso.inferior(function (lambda, self) {
145     if (!Espresso.isCallable(lambda)) {
146       throw new TypeError("{} is not callable.".format(lambda));
147     }
148 
149     return this.reduce(function (seive, v, k, t) {
150       if (lambda.call(self, v, k, t)) {
151         seive.push(v);
152       }
153       return seive;
154     }, []);
155   }),
156 
157   /** @function
158     @desc
159     Returns `true` if `lambda` returns `true` for every element
160     in the Enumerable, otherwise, it returns `false`.
161 
162     @param {Function} lambda The lambda that transforms an element in the enumerable.
163       @param {Object} lambda.value The value of the enumerated item.
164       @param {Object} lambda.key The key of the enumerated item.
165       @param {Object} lambda.self The object being enumerated over.
166     @param {Object} [self] The value of `this` inside the lambda.
167     @returns {Boolean} `true` if `lambda` returns `true` for every iteration.
168   */
169   every: Espresso.inferior(function (lambda, self) {
170     if (!Espresso.isCallable(lambda)) {
171       throw new TypeError("{} is not callable.".format(lambda));
172     }
173 
174     return this.reduce(function (every, v, k, t) {
175       return every && lambda.call(self, v, k, t);
176     }, true);
177   }),
178 
179   /** @function
180     @desc
181     Returns `true` if `lambda` returns `true` for at least one
182     element in the Enumerable, otherwise, it returns `false`.
183 
184     @param {Function} lambda The lambda that transforms an element in the enumerable.
185       @param {Object} lambda.value The value of the enumerated item.
186       @param {Object} lambda.key The key of the enumerated item.
187       @param {Object} lambda.self The object being enumerated over.
188     @param {Object} [self] The value of `this` inside the lambda.
189     @returns {Boolean} `true` if `lambda` returns `true` at least once.
190    */
191   some: Espresso.inferior(function (lambda, self) {
192     if (!Espresso.isCallable(lambda)) {
193       throw new TypeError("{} is not callable.".format(lambda));
194     }
195 
196     return this.reduce(function (every, v, k, t) {
197       return every || lambda.call(self, v, k, t);
198     }, false);
199   })
200 
201 };
