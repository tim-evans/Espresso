  3 /** @namespace
  4 
  5   The Enumerable mixin provides common operations
  6   on enumerations of objects.
  7 
  8   @requires `forEach`- the enumerator over the collection.
  9  */
 10 Espresso.Enumerable = /** @lends Espresso.Enumerable# */{
 11 
 12   /** @function
 13     @desc
 14     Iterates over the items on the Enumerable.
 15 
 16     The Function `forEach` should follow the specification as
 17     defined in the ECMAScript 5 standard. All function using
 18     `forEach` in the Enumerable mixin depend on it being this way.
 19 
 20     @param {Function} lambda The callback to call for each element.
 21       @param {Object} lambda.value The value of the enumerated item.
 22       @param {Object} lambda.key The key of the enumerated item.
 23       @param {Object} lambda.self The object being enumerated over.
 24     @param {Object} [self] The Object to use as this when executing the callback.
 25     @returns {void}
 26    */
 27   forEach: function (lambda, that) {
 28     throw new Error("You MUST override Espresso.Enumerable.forEach to be able " +
 29                     "to use the Enumerable mixin.");
 30   }.inferior(),
 31 
 32   /**
 33     Returns an array where each value on the enumerable
 34     is mutated by the lambda function.
 35 
 36     @param {Function} lambda The lambda that transforms an element in the enumerable.
 37       @param {Object} lambda.value The value of the enumerated item.
 38       @param {Object} lambda.key The key of the enumerated item.
 39       @param {Object} lambda.self The object being enumerated over.
 40     @param {Object} [self] The value of `this` inside the lambda.
 41     @returns {Array} The collection of results from the map function.
 42     @example
 43       var cube = function (n) { return n * n * n };
 44       alert([1, 2, 3, 4].map(cube));
 45       // -> [1, 8, 27, 64]
 46    */
 47   map: function (lambda, self) {
 48     var arr = [];
 49 
 50     // 4. If IsCallable(lambda) is false, throw a TypeError exception
 51     if (!Espresso.isCallable(lambda)) {
 52       throw new TypeError("{} is not callable.".fmt(lambda));
 53     }
 54 
 55     this.forEach(function (k, v) {
 56       arr.push(lambda.call(self, k, v, this));
 57     }, this);
 58     return arr;
 59   }.inferior(),
 60 
 61   /**
 62     Reduce the content of an enumerable down to a single value.
 63 
 64     @param {Function} lambda The lambda that performs the reduction.
 65       @param {Object} lambda.value The value of the enumerated item.
 66       @param {Object} lambda.key The key of the enumerated item.
 67       @param {Object} lambda.self The object being enumerated over.
 68     @param {Object} [seed] The seed value to provide for the first time.
 69     @returns {Object} The reduced output.
 70     @example
 71       var range = mix(Espresso.Enumerable, {
 72         begin: 0,
 73         end: 0,
 74 
 75         forEach: function (lambda, self) {
 76           var i = 0;
 77           for (var v = this.begin; v <= this.end; v++) {
 78             lambda.call(self, v, i++, this);
 79           }
 80         },
 81 
 82         create: function (begin, end) {
 83           return mix(this, { begin: begin, end: end }).into({});
 84         }
 85       }).into({});
 86 
 87       var multiply = function (a, b) { return a * b; };
 88       var factorial = function (n) {
 89         return range.create(1, n).reduce(multiply);
 90       }
 91 
 92       alert("5! is {}".fmt(factorial(5)));
 93       alert("120! is {}".fmt(factorial(120)));
 94    */
 95   reduce: function (lambda, seed) {
 96     var shouldSeed = (arguments.length === 1),
 97         self = this;
 98 
 99     // 4. If IsCallable(lambda) is false, throw a TypeError exception
100     if (!Espresso.isCallable(lambda)) {
101       throw new TypeError("{} is not callable.".fmt(lambda));
102     }
103 
104     this.forEach(function (v, k) {
105       if (shouldSeed) {
106         seed = v;
107         shouldSeed = false;
108       } else {
109         seed = lambda(seed, v, k, self);
110       }
111     });
112 
113     // 5. If len is 0 and seed is not present, throw a TypeError exception.
114     if (shouldSeed) {
115       throw new TypeError("There was nothing to reduce!");
116     }
117     return seed;
118   }.inferior(),
119 
120   /**
121     Converts an enumerable into an Array.
122 
123     @returns {Array} The enumerable as an Array.
124     @example
125       var range = mix(Espresso.Enumerable, {
126         begin: 0,
127         end: 0,
128 
129         forEach: function (lambda, self) {
130           var i = 0;
131           for (var v = this.begin; v <= this.end; v++) {
132             lambda.call(self, v, i++, this);
133           }
134         },
135 
136         create: function (begin, end) {
137           return mix(this, { begin: begin, end: end }).into({});
138         }
139       }).into({});
140 
141       alert(range.create(0, 200).toArray());
142       // -> [0, 1, 2, 3, 4, 5, ... 198, 199, 200]
143    */
144   toArray: function () {
145     return this.map(function (v) {
146       return v;
147     });
148   }.inferior(),
149 
150   /**
151     Returns all elements on the Enumerable for which the
152     input function returns true for.
153 
154     @param {Function} lambda The function to filter the Enumerable.
155       @param {Object} lambda.value The value of the enumerated item.
156       @param {Object} lambda.key The key of the enumerated item.
157       @param {Object} lambda.self The object being enumerated over.
158     @param {Object} [self] The value of `this` inside the lambda.
159     @returns {Object[]} An array with the values for which `lambda` returns `true`
160    */
161   filter: function (lambda, self) {
162     if (!Espresso.isCallable(lambda)) {
163       throw new TypeError("{} is not callable.".fmt(lambda));
164     }
165 
166     return this.reduce(function (seive, v, k, t) {
167       if (lambda.call(self, v, k, t)) {
168         seive.push(v);
169       }
170       return seive;
171     }, []);
172   }.inferior(),
173 
174   /**
175     Returns `true` if `lambda` returns `true` for every element
176     in the Enumerable, otherwise, it returns `false`.
177 
178     @param {Function} lambda The lambda that transforms an element in the enumerable.
179       @param {Object} lambda.value The value of the enumerated item.
180       @param {Object} lambda.key The key of the enumerated item.
181       @param {Object} lambda.self The object being enumerated over.
182     @param {Object} [self] The value of `this` inside the lambda.
183     @returns {Boolean} `true` if `lambda` returns `true` for every iteration.
184   */
185   every: function (lambda, self) {
186     if (!Espresso.isCallable(lambda)) {
187       throw new TypeError("{} is not callable.".fmt(lambda));
188     }
189 
190     return this.reduce(function (every, v, k, t) {
191       return every && lambda.call(self, v, k, t);
192     }, true);
193   }.inferior(),
194 
195   /**
196     Returns `true` if `lambda` returns `true` for at least one
197     element in the Enumerable, otherwise, it returns `false`.
198 
199     @param {Function} lambda The lambda that transforms an element in the enumerable.
200       @param {Object} lambda.value The value of the enumerated item.
201       @param {Object} lambda.key The key of the enumerated item.
202       @param {Object} lambda.self The object being enumerated over.
203     @param {Object} [self] The value of `this` inside the lambda.
204     @returns {Boolean} `true` if `lambda` returns `true` at least once.
205    */
206   some: function (lambda, self) {
207     if (!Espresso.isCallable(lambda)) {
208       throw new TypeError("{} is not callable.".fmt(lambda));
209     }
210 
211     return this.reduce(function (every, v, k, t) {
212       return every || lambda.call(self, v, k, t);
213     }, false);
214   }.inferior(),
215 
216   /**
217     Returns all values for the keys provided.
218 
219     @param {...} keys The keys to extract values from.
220     @returns {Object[]} The values for the keys provided (not.
221    */
222   extract: function () {
223     var keys = Espresso.toArray(arguments);
224     
225     return this.filter(function (v, k) {
226       return keys.indexOf(k) !== -1;
227     });
228   },
229 
230   /**
231     Returns the first value for which `lambda` returns `true`.
232     If nothing is found, `find` will return `ifnone`, a default
233     value provided as an optional argument or `undefined` if
234     `ifnone` was not provided.
235 
236     @param {Function} lambda The lambda that returns something truthy or falsy.
237       @param {Object} lambda.value The value of the enumerated item.
238       @param {Object} lambda.key The key of the enumerated item.
239       @param {Object} lambda.self The object being enumerated over.
240     @param {Object} [ifnone] The value to return if nothing is found.
241     @returns {Object} The first object to which `lambda` returns something truthy.
242    */
243   find: function (lambda, ifnone) {
244     if (!Espresso.isCallable(lambda)) {
245       throw new TypeError("{} is not callable.".fmt(lambda));
246     }
247 
248     var finished = false;
249     return this.reduce(function (result, v, k, that) {
250       if (!finished && lambda(v, k, that)) {
251         finished = true;
252         result = v;
253       }
254       return result;
255     }, ifnone);
256   },
257 
258   /**
259     Whether or not the {@link Espresso.Enumerable} contains
260     the variables.
261 
262     @param {...} values The values to check whether they exist
263       on the Enumerable.
264    */
265   contains: function (val) {
266     var args = Espresso.toArray(arguments);
267 
268     if (args.length > 1) {
269       return args.every(function (v, k) {
270         return this.contains(v);
271       }, this);
272     } else {
273       return this.reduce(function (contained, v, k) {
274         return contained || v === val;
275       }, false);
276     }
277   }
