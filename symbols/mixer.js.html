  3 /** @function
  4   @desc
  5 
  6   `mix` provides a way to combine arbritrary objects together.
  7 
  8   The combination can be as simple as adding the properties on
  9   an object onto another:
 10 
 11       var Caffeinated = { isCaffeinated: true };
 12       var Coffee = mix({
 13         isDecaf: function () {
 14           return !!this.isCaffeinated;
 15         }
 16       }).into({});
 17 
 18       decaf = mix(Coffee).into({});
 19       decaf.isDecaf();
 20       // -> true
 21 
 22       caf = mix(Caffeinated, Coffee).into({});
 23       caf.isDecaf();
 24       // -> false
 25 
 26   `mix` takes this a bit furthur, allowing properties on the
 27   objects being mixed in to be altered at mixin time using
 28   Espresso's decorator API.
 29 
 30   The API hook is adding an underscore (`_`) hash with a
 31   function that can change the decorated object in place by
 32   returning the new desired value. For examples on how to use
 33   the decorator API, look at the `alias` and `inferior` for
 34   general purpose decorators and `refine` for a funnction
 35   decorator.
 36 
 37   Using `mix`, you can design an Object-Oriented `Class`
 38   object with while still inheriting all of the decorators
 39   that `mix` applies:
 40 
 41       Class = mix({
 42         extend: (function () {
 43           var initializing = false;
 44 
 45           return function () {
 46             initializing = true;
 47             var prototype = new this();
 48             initializing = false;
 49 
 50             mix.apply(null, Espresso.A(arguments)).into(prototype);
 51 
 52             function Class() {
 53               if (!initializing && Espresso.isCallable(this.init)) {
 54                 this.init.apply(this, arguments);
 55               }
 56             }
 57 
 58             Class.prototype = prototype;
 59             Class.constructor = Class;
 60             Class.extend = arguments.callee;
 61             return Class;
 62           };
 63         }())
 64       }).into(function () {});
 65 
 66   @param {...} mixins Objects to mixin to the target provided on into.
 67   @returns {Object} An object with `into` field, call into with the target
 68                     to apply the mixins on. That will return the target
 69                     with the mixins on it.
 70  */
 71 mix = function () {
 72   var mixins = arguments,
 73       i = 0, len = mixins.length;
 74 
 75   return {
 76     into: function (target) {
 77       var mixin, key, value, _, decorator;
 78 
 79       if (target == null) {
 80         throw new TypeError("Cannot mix into null or undefined values.");
 81       }
 82 
 83       for (; i < len; i += 1) {
 84         mixin = mixins[i];
 85         for (key in mixin) {
 86           value = mixin[key];
 87 
 88           // Function annotation API
 89           _ = value && value._;
 90           if (_ != null) {
 91             for (decorator in _) {
 92               if (_.hasOwnProperty(decorator)) {
 93                 value = _[decorator](target, value, key);
 94               }
 95             }
 96           }
 97           if (typeof value !== "undefined") target[key] = value;
 98         }
 99 
100         // Take care of IE clobbering `toString` and `valueOf`
101         if (mixin && mixin.toString !== Object.prototype.toString) {
102           target.toString = mixin.toString;
103         } else if (mixin && mixin.valueOf !== Object.prototype.valueOf) {
104           target.valueOf = mixin.valueOf;
105         }
106       }
107       return target;
108     }
109   };
110 };
111 
112 // Apply it at the global scope
113 this.mix = mix;
