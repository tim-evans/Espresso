  5   /**
  6     Currying transforms a function by transforming by composing it into
  7     functions that each take a subset of the arguments of the whole.
  8     The aggregate of all of the arguments passed into curried function
  9     and the actual function call is what will be provided to the function
 10     that's being curried.
 11 
 12     This effectively means that you can transform a very simple call into
 13     atomic calls as follows:
 14 
 15         var mult = function () {
 16           return Espresso.A(arguments).reduce(function (E, x) { return E * x; }, 1);
 17         };
 18 
 19         alert(mult.curry(2, 2)());
 20         // => 4
 21 
 22     Or specify a function that is a subset of the first:
 23 
 24         var add = function () {
 25           return Espresso.A(arguments).reduce(function (E, x) { return E + x; }, 0);
 26         };
 27 
 28         var inc = add.curry(1);
 29 
 30         alert(inc(5));
 31         // => 6
 32 
 33     In layman's terms, `curry` will prepopulate arguments for a function.
 34     @param {...} args Arguments that will be curried to the function.
 35     @returns {Function} The function with the arguments provided saved for later.
 36    */
 37   curry: function () {
 38     var Target, A;
 39 
 40     // 1. Let Target be the this value.
 41     Target = this;
 42 
 43     // 2. If IsCallable(Target) is false, throw a TypeError exception
 44     if (!Espresso.isCallable(Target)) {
 45       throw new TypeError("The Target is not callable.");
 46     }
 47 
 48     // 3. Let A be a new (possibly empty) internal list of
 49     //    all argument values (arg1, arg2, etc), in order
 50     A = Espresso.A(arguments);
 51 
 52     return function () {
 53       return Target.apply(this, A.concat(Espresso.A(arguments)));
 54     };
 55   },

 57   /** @function
 58     @desc

 90     }

 93     //    all argument values provided after self
 94     //    (arg1, arg2, etc), in order

 99       if (this instanceof bound) {

101         // When the [[Construct]] internal method of a function object, F,
102         // that was created using the bind function is called with a list of
103         // arguments ExtraArgs, the following steps are taken:
104 
105         // 1. Let the target be the value of F's [[TargetFunction]] internal property.
106         // 2. If target has no [[Construct]] internal method, a TypeError exception is thrown.
107         // 3. Let boundArgs be the value of F's [[BoundArgs]] internal property.
108         // 4. Left args be a new list containing the same values as the list boundArgs in the same order followed by the same values as the list ExtraArgs in the same order.
109         // 5. Return the result of calling the [[Construct]] internal method of target providing args as the arguments.

111         Type.prototype = Target.prototype;
112         that = new Type();
113

115         return that;
116       } else {
117         // 15.3.4.5.1 [[Call]]
118         // When the [[Call]] internal method of a function object, F,
119         // which was created using the bind function is called with a this
120         // value and a list of arguments ExtraArgs, the following steps are taken:
121         // 1. Let boundArgs be the value of F's [[BoundArgs]] internal property.
122         // 2. Let boundThis be the value of F's [[BoundThis]] internal property.
123         // 3. Let target be the value of F's [[TargetFunction]] internal property.
124         return Target.apply(self, A.concat(Espresso.A(arguments)));
125       }
