  2 /** @namespace
  3 
  4   [Key-Value Observing][kvo] (KVO) is a mechanism that allows
  5   objects to be notified of changes to specified properties of
  6   other Objects. It is based off of the observer pattern, which
  7   in turn is built on top of the Publish-Subscribe pattern.
  8 
  9   KVO is used on top of {@link Espresso.Subscribable} for notifying
 10   observers that a change occured.
 11 
 12   To understand Key-Value coding, you must understand property
 13   paths first. This simply means that you need to understand
 14   the Object model of the object that you are doing a `get` or
 15   `set` on. Take the following example:
 16 
 17       var Beatles = mix(Espresso.Observable).into({
 18         Paul: {
 19           instruments: ['vocals', 'bass', 'guitar', 'piano',
 20                         'keyboards', 'drums', 'ukelele',
 21                         'mandolin']
 22         },
 23         John: {
 24           instruments: ['vocals', 'guitar', 'piano', 'banjo',
 25                         'harmonica', 'mellotron',
 26                         'six-string bass', 'percussion']
 27         },
 28         Ringo: {
 29           instruments: ['drums', 'vocals', 'percussion',
 30                         'tambourine']
 31         },
 32         George: {
 33           instruments: ['guitar', 'vocals', 'bass', 'keyboards',
 34                         'ukelele', 'mandolin', 'sitar', 'tambura',
 35                         'sarod', 'swarmandal']
 36         }
 37       });
 38 
 39       alert(Beatles.getPath('Paul.instruments.0'));
 40       // => 'vocals'
 41 
 42   Using `get` provides optimizations such as caching on an Object.
 43 
 44   Using `set` provides notifications to observing functions /
 45   properties.
 46 
 47   The Observable mixin provides the ability to have dynamically computed
 48   properties via the `property` decorator on functions and the
 49   ability to intercept `get`s or `set`s to unknown properties via
 50   `unknownProperty`.
 51 
 52   Computed properties are simply a function that takes 2 arguments,
 53   the key and the value of the property that triggered the function
 54   call. These properties may also have dependent keys. When a
 55   property has dependent keys, every single time a dependent key
 56   gets `set`, the property will get recomputed.
 57 
 58   Consider the following:
 59 
 60       var Box = mix(Espresso.Observable).into({
 61         width: 0,
 62         height: 0,
 63         depth: 0,
 64 
 65         volume: Espresso.property(function () {
 66           return this.get('width') * this.get('height') * this.get('depth');
 67         }, 'width', 'height', 'depth').cacheable()
 68       });
 69 
 70   The `volume` property will get recomputed every single time the
 71   `width`, `height`, or `depth` values change. If you had another
 72   object that you would like to monitor the changes, perhaps a
 73   renderer, you could attach observers to each of the properties
 74   by subscribing to the property path (via
 75   {@link Espresso.Subscribable#subscribe}), providing any property paths
 76   that you would like to be notified on.
 77 
 78     [kvo]: http://developer.apple.com/library/mac/#documentation/Cocoa/Conceptual/KeyValueObserving/KeyValueObserving.html
 79 
 80   @extends Espresso.Subscribable
 81  */
 82 Espresso.Observable = mix(Espresso.Subscribable).into(/** @lends Espresso.Observable# */{

 84   /**
 85     Walk like a duck.
 86     @type Boolean
 87    */
 88   isObservable: true,

 90   /**
 91     Initialize the observer. This needs to be explicitly
 92     called to activate property observing.
 93 
 94     When creating your base object for your library, you
 95     should use the following boilerplate to make property
 96     observing automatically initialize (with the following
 97     boilerplate assuming your constructor is called `init`):
 98 
 99         mix({
100           init: function (original) {
101             this.initObservable();
102             return original.apply(null, Espresso.A(arguments).slice(1));
103           }.refine()
104         }).into(Espresso.Observable);
105 
106     @returns {void}
107    */
108   initObservable: function () {
109     if (this.__isObservableInitialized__) { return; }
110     this.__isObservableInitialized__ = true;
111 
112     var key, property, i = 0, len, dependents, meta = this.__espmeta__,
113         dependent, iDependent, object, notifier;
114 
115     /** @ignore */
116     notifier = function (key) {
117       this.set(key);
118     };
119 
120     for (key in meta) { // Iterate over all keys
121       if (meta.hasOwnProperty(key) && meta[key].referenceKey) {
122         property = this[key];
123 
124         if (Espresso.isCallable(property) &&
125             property.isProperty && property.dependentKeys) {
126 
127           dependents = property.dependentKeys;
128           len = dependents.length;
129           for (i = 0; i < len; i += 1) {
130             dependent = dependents[i];
131             object = this;
132 
133             // If it's a property path, follow the chain.
134             if (dependent.indexOf('.') !== -1) {
135               iDependent = dependent.lastIndexOf('.');
136               object = Espresso.getObjectFor(dependent.slice(0, iDependent));
137               dependent = dependent.slice(iDependent + 1);
138             }
139 
140             // Subscribe to the events.
141             if (object && object.isObservable && object.isSubscribable) {
142               object.subscribe(dependent, notifier.bind(this, key), { synchronous: true });
143             }
144           }
145         }
146       }
147     }
148   },

150   /**
151     Get a value on an object that is a property path.
152 
153     This function will return the value given the
154     property path using `get` when necessary.
155 
156     This means you should write:
157 
158         zombie.getPath('brain.isDelicious');
159 
160     instead of:
161 
162         zombie.get('brain.isDelicious');
163 
164     @param {String} key The property path to lookup on the object.
165     @returns {Object} The value of the key.
166    */
167   getPath: function (k) {
168     k = k.toString();
169 
170     var result = this, key = k, idx;
171     while (result != null && k.length > 0) {
172       idx = k.indexOf('.');
173       key = (idx === -1) ? k : k.slice(0, idx);
174       k = (idx === -1) ? "" : k.slice(idx + 1);
175       result = result.get ? result.get(key) : result[key];
176     }
177     return result;
178   },

180   /**
181     Get a value on an object.
182 
183     Use this instead of subscript (`[]`) or dot notation
184     for public variables. Otherwise, you won't reap benefits
185     of being notified when they are set, or if the property
186     is computed.
187 
188     Get is tolerant of when trying to access objects that
189     don't exist- it will return undefined in that case.
190 
191     @param {String} key The key to lookup on the object.
192     @returns {Object} The value of the key.
193    */
194   get: function (k) {
195     k = k.toString();
196     var object = this, value, info, refKey;
197 
198     // Retrieve metadata about this property
199     info = this.__espmeta__ ? this.__espmeta__[k] : null;
200     refKey = k;
201     if (info) {
202       if (info.closureKey) {
203         k = info.closureKey;
204       }
205 
206       if (info.referenceKey && info.isComputed) {
207         refKey = info.referenceKey;
208       }
209     }
210 
211     value = object[k];
212     // Deal with properties
213     if (value && value.isProperty) {
214       // If the value of the property is cached,
215       // retrieve it from the cache and return it.
216       if (value.isCacheable) {
217         object.__cache__ = object.__cache__ || {};
218         if (!object.__cache__.hasOwnProperty(k)) {
219           object.__cache__[k] = value.call(object, refKey);
220         }
221         value = object.__cache__[k];
222 
223       // Otherwise, we need to retrieve the value
224       } else {
225         value = value.call(object, refKey);
226       }
227 
228     // Unknown properties
229     } else if (typeof value === "undefined") {
230       value = object.unknownProperty(k);
231     }
232     return value;
233   },

235   /**
236     Set a value that is a property path.
237 
238     This function will return the value given the
239     property path using `set` and `get` when necessary.
240 
241     This means you should write:
242 
243         zombie.setPath('brain.isDelicious', true);
244 
245     instead of:
246 
247         zombie.set('brain.isDelicious', true);
248 
249     @param {String} key The property path to lookup on the object.
250     @param {Object} value The value to set the object at the key's path to.
251     @returns {Object} The reciever.
252    */
253   setPath: function (k, v) {
254     k = k.toString();
255 
256     var idx = k.lastIndexOf('.'),
257         object = idx === -1 ? this : this.getPath(k.slice(0, idx));
258 
259     if (idx !== -1) k = k.slice(idx + 1);
260     object && object.set ? object.set(k, v) : object[k] = v;
261     return this;
262   },

264   /**
265     Set a value on an object.
266 
267     Use this instead of subscript (`[]`) or dot notation
268     for public variables. Otherwise, you won't reap benefits
269     of being notified when they are set, or if the property
270     is computed.
271 
272     Set is tolerant of when trying to access objects that
273     don't exist- it will ignore your attempt in that case.
274 
275     @param {String} key The key to lookup on the object.
276     @param {Object} value The value to set the object at the key's path to.
277     @returns {Object} The reciever.
278    */
279   set: function (k, v) {
280     k = k.toString();
281 
282     var property, key = k, value = v, idx = key.lastIndexOf('.'), object = this,
283         result, didChange = false, info, refKey, isComputed = false;
284 
285     // Retrieve metadata about this property
286     info = this.__espmeta__ ? this.__espmeta__[k] : null;
287     refKey = key;
288     if (info) {
289       if (info.closureKey) {
290         key = info.closureKey;
291       }
292 
293       if (info.referenceKey) {
294         refKey = info.referenceKey;
295       }
296       isComputed = info.isComputed;
297     }
298 
299     property = object[key];
300 
301     // Properties first.
302     if (property && property.isProperty) {
303       // If setting it multiple times with the same
304       // value does nothing, check to see if we should
305       // set it again.
306       if (property.isIdempotent) {
307         object.__value__ = object.__value__ || {};
308         if (object.__value__[key] !== value) {
309           result = property.call(object, refKey, value);
310           didChange = true;
311         }
312         object.__value__[key] = value;
313 
314       // Otherwise, call the property with the key and value.
315       } else {
316         result = property.call(object, refKey, value);
317         didChange = true;
318       }
319 
320       // Cache the new return from the function
321       // so we don't have to waste another lookup later.
322       if (property.isCacheable && didChange) {
323         object.__cache__ = object.__cache__ || {};
324         object.__cache__[key] = result;
325       }
326 
327     // Unknown property
328     } else if (typeof property === "undefined") {
329       this.unknownProperty(k, v);
330 
331     // Simply set it (taking into account computed ECMAScript5 properties).
332     } else {
333       object[isComputed ? refKey : key] = value;
334     }
335 
336     // Expected behaviour is strange unless publishes
337     // are done immediately.
338     if (object.publish && !(property && property.isIdempotent && !didChange)) {
339       object.publish(refKey, value);
340     }
341 
342     return this;
343   },
344 
345   /**
346     Called whenever you try to get or set a nonexistent
347     property.
348 
349     This is a generic property that you can override to
350     intercept general gets and sets, making use out of them.
351 
352     @param {String} key The unknown key that was looked up.
353     @param {Object} [value] The value to set the key to.
354     @returns {Object} The value of the key.
355    */
356   unknownProperty: function (key, value) {
357     if (arguments.length === 2) {
358       var parts = key.split('.'), part, root = this,
359           len = parts.length - 1, i = 0, o;
360       for (; i < len; i++) {
361         part = parts[i];
362         o = root.get ? root.get(part) : Espresso.getObjectFor(part, root);
363 
364         // Don't mess with existing objects.
365         if (typeof o !== "undefined") {
366           root = o;
367 
368         // Create new Objects when they don't exist
369         } else {
370           root[part] = {};
371           root = root[part];
372         }
373       }
374 
375       part = parts[len];
376       if (root[part] != null && root.set) {
377         root.set(part, value);
378       } else {
379         root[part] = value;
380       }
381     }
382     return Espresso.getObjectFor(key, this);
383   }
