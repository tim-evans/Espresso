  2 /** @namespace
  3 
  4   [Key-Value Observing][kvo] (KVO) is a mechanism that allows
  5   objects to be notified of changes to specified properties of
  6   other Objects. It is based off of the observer pattern, which
  7   in turn is built on top of the Publish-Subscribe pattern.
  8 
  9   KVO is used on top of {@link Espresso.Subscribable} for notifying
 10   observers that a change occured.
 11 
 12   To understand Key-Value coding, you must understand property
 13   paths first. This simply means that you need to understand
 14   the Object model of the object that you are doing a `get` or
 15   `set` on. Take the following example:
 16 
 17       var Beatles = mix(Espresso.Observable).into({
 18         Paul: {
 19           instruments: ['vocals', 'bass', 'guitar', 'piano',
 20                         'keyboards', 'drums', 'ukelele',
 21                         'mandolin']
 22         },
 23         John: {
 24           instruments: ['vocals', 'guitar', 'piano', 'banjo',
 25                         'harmonica', 'mellotron',
 26                         'six-string bass', 'percussion']
 27         },
 28         Ringo: {
 29           instruments: ['drums', 'vocals', 'percussion',
 30                         'tambourine']
 31         },
 32         George: {
 33           instruments: ['guitar', 'vocals', 'bass', 'keyboards',
 34                         'ukelele', 'mandolin', 'sitar', 'tambura',
 35                         'sarod', 'swarmandal']
 36         }
 37       });
 38 
 39       alert(Beatles.get('Paul.instruments.0'));
 40       // => 'vocals'
 41 
 42   Using `get` provides optimizations such as caching on an Object.
 43 
 44   Using `set` provides notifications to observing functions /
 45   properties.
 46 
 47   The Observable mixin provides the ability to have dynamically computed
 48   properties via the `property` decorator on functions and the
 49   ability to intercept `get`s or `set`s to unknown properties via
 50   `unknownProperty`.
 51 
 52   Computed properties are simply a function that takes 2 arguments,
 53   the key and the value of the property that triggered the function
 54   call. These properties may also have dependent keys. When a
 55   property has dependent keys, every single time a dependent key
 56   gets `set`, the property will get recomputed.
 57 
 58   Consider the following:
 59 
 60       var Box = mix(Espresso.Observable).into({
 61         width: 0,
 62         height: 0,
 63         depth: 0,
 64 
 65         volume: function () {
 66           return this.get('width') * this.get('height') * this.get('depth');
 67         }.property('width', 'height', 'depth').cacheable()
 68       });
 69 
 70   The `volume` property will get recomputed every single time the
 71   `width`, `height`, or `depth` values change. If you had another
 72   object that you would like to monitor the changes, perhaps a
 73   renderer, you could attach observers to each of the properties
 74   by subscribing to the property path (via
 75   {@link Espresso.Subscribable#subscribe}), providing any property paths
 76   that you would like to be notified on.
 77 
 78     [kvo]: http://developer.apple.com/library/mac/#documentation/Cocoa/Conceptual/KeyValueObserving/KeyValueObserving.html
 79 
 80   @extends Espresso.Subscribable
 81  */
 82 Espresso.Observable = mix(Espresso.Subscribable).into(/** @lends Espresso.Observable# */{
 83 
 84   /**
 85     Get a value on an object.
 86 
 87     Use this instead of subscript (`[]`) or dot notation
 88     for public variables. Otherwise, you won't reap benefits
 89     of being notified when they are set, or if the property
 90     is computed.
 91 
 92     Get is tolerant of when trying to access objects that
 93     don't exist- it will return undefined in that case.
 94 
 95     @param {String} key The key to lookup on the object.
 96     @returns {Object} The value of the key.
 97    */
 98   get: function (k) {
 99     k = k.toString();
100     var key = k, value, idx = key.lastIndexOf('.'), object;
101     if (idx === -1) {
102       object = this;
103     } else {
104       object = Espresso.getObjectFor(key.slice(0, idx), this);
105       key = key.slice(idx + 1);
106     }
107 
108     if (object) {
109       value = object[key];
110       if (typeof value === "undefined") {
111         if (Espresso.isCallable(object.unknownProperty)) {
112           value = object.unknownProperty.call(object, key);
113         } else {
114           value = this.unknownProperty(k);
115         }
116       } else if (value && value.isProperty) {
117         if (value.isCacheable) {
118           object.__cache__ = object.__cache__ || {};
119           if (!object.__cache__.hasOwnProperty(key)) {
120             object.__cache__[key] = value.call(object, key);
121           }
122           return object.__cache__[key];
123         }
124         value = value.call(object, key);
125       }
126       return value;
127     }
128     return this.unknownProperty(k);
129   },
130 
131   /**
132     Set a value on an object.
133 
134     Use this instead of subscript (`[]`) or dot notation
135     for public variables. Otherwise, you won't reap benefits
136     of being notified when they are set, or if the property
137     is computed.
138 
139     Set is tolerant of when trying to access objects that
140     don't exist- it will ignore your attempt in that case.
141 
142     @param {String} key The key to lookup on the object.
143     @param {Object} value The value to set the object at the key's path to.
144     @returns {Object} The reciever.
145    */
146   set: function (k, v) {
147     k = k.toString();
148 
149     var property, key = k, value = v, idx = key.lastIndexOf('.'), object, result, didChange = false;
150     if (idx === -1) {
151       object = this;
152     } else {
153       object = Espresso.getObjectFor(key.slice(0, idx), this);
154       key = key.slice(idx + 1);
155     }
156 
157     if (object) {
158       property = object[key];
159 
160       if (property && property.isProperty) {
161         if (property.isIdempotent) {
162           object.__value__ = object.__value__ || {};
163           if (object.__value__[key] !== value) {
164             result = property.call(object, key, value);
165             didChange = true;
166           }
167           object.__value__[key] = value;
168         } else {
169           result = property.call(object, key, value);
170           didChange = true;
171         }
172 
173         if (property.isCacheable && didChange) {
174           object.__cache__ = object.__cache__ || {};
175           object.__cache__[key] = result;
176         }
177       } else if (typeof property === "undefined") {
178         if (Espresso.isCallable(object.unknownProperty)) {
179           object.unknownProperty.call(object, key, value);
180         } else {
181           this.unknownProperty(k, v);
182         }
183       } else {
184         object[key] = value;
185       }
186 
187       // Expected behaviour is strange unless publishes
188       // are done immediately.
189       if (object.publish && !(property && property.isIdempotent && !didChange)) {
190         object.publish(key, value);
191       }
192     } else {
193       this.unknownProperty(k, v);
194     }
195     return this;
196   },
197 
198   /**
199     Called whenever you try to get or set a nonexistent
200     property.
201 
202     This is a generic property that you can override to
203     intercept general gets and sets, making use out of them.
204 
205     @param {String} key The unknown key that was looked up.
206     @param {Object} [value] The value to set the key to.
207     @returns {Object} The value of the key.
208    */
209   unknownProperty: function (key, value) {
210     if (typeof value !== "undefined") {
211       this[key] = value;
212     }
213     return this[key];
214   }
215
