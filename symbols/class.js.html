  3 /** @class
  4 
  5   Class provides a manner to perform classical inheritance
  6   with the ability to mixin modules and extend current classes.
  7 
  8   Class is {@link Espresso.Template}'s sister, which provides
  9   a root class for all other classes to inherit from.
 10 
 11   This means that the following will work:
 12 
 13       var Person = Espresso.Class.extend({
 14         init: function (isDancing) {
 15           this.dancing = isDancing;
 16         },
 17 
 18         isDancing: function () {
 19           return this.get('dancing');
 20         }.property('dancing')
 21       });
 22 
 23       var Ninja = Person.extend({
 24         init: function ($super) {
 25           $super(false);
 26         }.around(),
 27 
 28         hasShuriken: function () {
 29           return true; // of course!
 30         }.property()
 31       });
 32 
 33       var p = new Person(true);
 34       alert(p.get('isDancing'));
 35       // => true
 36 
 37       var n = new Ninja();
 38       alert(n.get('isDancing'));
 39       // => false
 40       alert(n.get('hasShuriken'));
 41       // => true
 42 
 43       alert(p instanceof Person && p instanceof Espresso.Class &&
 44             n instanceof Ninja && n instanceof Person && n instanceof Espresso.Class);
 45       // => true
 46 
 47   Based off of John Resig's [simple inheritance][resig].
 48 
 49     [resig]: http://ejohn.org/blog/simple-javascript-inheritance/
 50 
 51   @extends Espresso.PubSub
 52   @extends Espresso.KVO
 53  */
 54 Espresso.Class = function () {};
 55 
 56 mix(/** @scope Espresso.Class */{
 57 
 58   /** @function
 59     @desc
 60 
 61     Extend the class with the given properties.
 62     Multiple inheritance is not allowed without
 63     breaking the inheritance chain.
 64 
 65     @returns {Espresso.Class} The extended Class.
 66    */
 67   extend: (function () {
 68     var initializing = false;
 69 
 70     return function () {
 71       // Prevent initialization when creating the Class.
 72       initializing = true;
 73       var prototype = new this(), i, len = arguments.length;
 74       initializing = false;
 75 
 76       for (i = 0; i < len; i += 1) {
 77         mix.apply(null, [arguments[i]]).into(prototype);
 78       }
 79 
 80       function Class() {
 81         if (!initializing && Espresso.isCallable(this.init)) {
 82           this.init.apply(this, arguments);
 83         }
 84       }
 85 
 86       Class.prototype = prototype;
 87       Class.constructor = Class;
 88 
 89       Class.extend = arguments.callee;
 90       return Class;
 91     };
 92   }())
 93 }).into(Espresso.Class);
 94 
 95 mix(Espresso.PubSub, Espresso.KVO, /** @scope Espresso.Class.prototype */{
 96 
 97   /**
 98     Filters out private variables and functions
 99     when serializing the JSON to a String.
100 
101     @returns {Object} The object hash to use when converting to JSON.
102    */
103   toJSON: function (key) {
104     var k, v, json = {};
105     for (k in this) {
106       v = this.get(k);
107       if (k.charAt(0) !== "_" && !Espresso.isCallable(v) && k !== 'unknownProperty') {
108         json[k] = v;
109       }
110     }
111     return json;
112   }
113 }).into(Espresso.Class.prototype);
