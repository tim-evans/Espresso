 21   /** @function
 22     @desc
 23     Date Formatting support (for use with `format`).
 24 
 25     The following flags are acceptable in a format string:
 26 
 27      - `a` The abbreviated weekday name ("Sun")
 28      - `A` The full weekday name ("Sunday")
 29      - `b` The abbreviated month name ("Jan")
 30      - `B` The full month name ("January")
 31      - `c` The preferred local date and time representation
 32      - `d` Day of the month (01..31)
 33      - `H` Hour of the day, 24-hour clock (00..23)
 34      - `I` Hour of the day, 12-hour clock (01..12)
 35      - `j` Day of the year (001..366)
 36      - `m` Month of the year (01..12)
 37      - `M` Minute of the hour (00..59)
 38      - `p` Meridian indicator ("AM" or "PM")
 39      - `S` Second of the minute (00..60)
 40      - `U` Week number of the current year, starting with the first Sunday as the first day of the first week (00..53)
 41      - `W` Week number of the current year, starting with the first Monday as the first day of the first week (00..53)
 42      - `w` Day of the week (Sunday is 0, 0..6)
 43      - `x` Preferred representation for the date alone, no time
 44      - `X` Preferred representation for the time alone, no date
 45      - `y` Year without a century (00..99)
 46      - `Y` Year with century
 47 
 48     For example:
 49 
 50         alert("Today is {:A, B d, Y}.".format(new Date()));
 51 
 52         alert("The time is: {:c}.".format(new Date()));
 53 
 54     Note: all times used with `format` are **NOT** in UTC.
 55 
 56     @param {String} spec The specifier to transform the date to a formatted string.
 57     @returns {String} The Date transformed into a string as specified.
 58    */
 59   __format__: (function () {
 60     return function (spec) {
 61       var result = [], i = 0;
 62 
 63       for (; i < spec.length; i += 1) {
 64         switch (spec.charAt(i)) {
 65         case 'a':
 66           result[result.length] = Date.days[this.getDay()].slice(0, 3);
 67           break;
 68         case 'A':
 69           result[result.length] = Date.days[this.getDay()];
 70           break;
 71         case 'b':
 72           result[result.length] = Date.months[this.getMonth()].slice(0, 3);
 73           break;
 74         case 'B':
 75           result[result.length] = Date.months[this.getMonth()];
 76           break;
 77         case 'c':
 78           result[result.length] = "{0:a b} {1:2} {0:H:M:S Y}".format(this, this.getDate());
 79           break;
 80         case 'd':
 81           result[result.length] = "{:02}".format(this.getDate());
 82           break;
 83         case 'H':
 84           result[result.length] = "{:02}".format(this.getHours());
 85           break;
 86         case 'I':
 87           result[result.length] = "{:02}".format(this.getHours() % 12);
 88           break;
 89         case 'j':
 90           result[result.length] = "{:03}".format(Math.ceil((this - new Date(this.getFullYear(), 0, 1)) / 86400000));
 91           break;
 92         case 'm':
 93           result[result.length] = "{:02}".format(this.getMonth() + 1);
 94           break;
 95         case 'M':
 96           result[result.length] = "{:02}".format(this.getMinutes());
 97           break;
 98         case 'p':
 99           result[result.length] = this.getHours() > 11 ? "PM" : "AM";
100           break;
101         case 'S':
102           result[result.length] = "{:02}".format(this.getSeconds());
103           break;
104         case 'U':
105           // Monday as the first day of the week
106           var day = ((this.getDay() + 6) % 7) + 1;
107           result[result.length] = "{:02}".format(
108             Math.ceil((((this - new Date(this.getFullYear(), 0, 1)) / 86400000) + day) / 7) - 1);
109           break;
110         case 'w':
111           result[result.length] = this.getDay();
112           break;
113         case 'W':
114           result[result.length] = "{:02}".format(
115             Math.ceil((((this - new Date(this.getFullYear(), 0, 1)) / 86400000) + this.getDay() + 1) / 7) - 1);
116           break;
117         case 'x':
118           result[result.length] = "{:m/d/y}".format(this);
119           break;
120         case 'X':
121           result[result.length] = this.toLocaleTimeString();
122           break;
123         case 'y':
124           result[result.length] = "{:02}".format(this.getYear() % 100);
125           break;
126         case 'Y':
127           result[result.length] = this.getFullYear();
128           break;
129         default:
130           result[result.length] = spec.charAt(i);
131         }
132       }
133       return result.join('');
134     };
135   }())
136 }).into(Date.prototype);
