  5   Espresso is a JavaScript library to be used as a
  6   foundation library to create JavaScript libraries.
  7   This library is made with the to aid in creating

 24   /** @function
 25     @desc
 26 
 27     Lookup a variable's value given its Object notation.
 28     This requires absolute queries to the Object, using
 29     idiomatic JavaScript notation.
 30 
 31     @example
 32       // No scope assumes the object has is at the global scope.
 33       window.environment = {
 34         isBrowser: (function () {
 35           return document in this;
 36         }())
 37       };
 38 
 39       alert(Espresso.getObjectFor("environment.isBrowser"));
 40 
 41     @example
 42       alert(Espresso.getObjectFor("lang.pr._coffee", {
 43         lang: {
 44           en: { _coffee: "coffee" },
 45           pr: { _coffee: "cafe" }
 46         }
 47       }));
 48       // -> "cafe"
 49 
 50     @example
 51       alert(Espresso.getObjectFor("options[0]", {
 52         options: ["espresso", "coffee", "tea"]
 53       }));
 54       // -> "espresso"
 55 
 56     @param {String} key The key to get on the target.
 57     @param {Object} [object] The target object to get a value from.
 58     @returns {Object} The referenced value in the args passed in.
 59    */
 60   getObjectFor: (function () {
 61     /** @ignore */
 62     // Return the property or `undefined`
 63     var getProperty = function (property, obj) {
 64       if (property in obj) {
 65         obj = obj[property];
 66       } else {
 67         obj = void 0;
 68       }
 69       return obj;
 70     }, G = this,
 71     arrError = "Expected EOS, '.', or '[' as the next character in the property path, but got '{}'\n{}\n{: >{}}",
 72     pathError = "Expected non-delimiter character (anything but '[' or '.')  in the property path, but got '{}'\n{}\n{: >{}}";
 73 
 74     return function (key, object) {
 75       var iattr = key.indexOf('.'),
 76           iarr = key.indexOf('['),
 77           fullKey = key, idx, chr;
 78 
 79       // Use global scope as default
 80       object = (arguments.length === 1) ? G: object;
 81 
 82       // Nothing to look up on undefined or null objects.
 83       if (object == null) {
 84         return object;
 85       }
 86 
 87       // Array accessor (`[]`) access
 88       if ((iarr < iattr || iattr === -1) && iarr > -1) {
 89 
 90         // Found something that looks like `ingredients[0]`
 91         // Unpack the first part, then deal with the array subscript.
 92         if (iarr !== 0) {
 93           object = getProperty(key.slice(0, iarr), object);
 94           key = key.slice(iarr + 1);
 95           idx = iarr;
 96         }
 97 
 98         iarr = key.indexOf(']');
 99 
100         // Unpack the property
101         object = getProperty(key.slice(0, iarr), object);
102 
103         // Eat the rest of the descriptor
104         key = key.slice(iarr + 1);
105         idx += iarr + 2;
106 
107         chr = key.charAt(0);
108         // Malformed property path
109         if (!(chr === "" || chr === '.' || chr === '[')) {
110           throw new Error(Espresso.format(arrError, chr, fullKey, idx + 1, '^'));
111         }
112 
113         // Eat up the dot.
114         if (key.length && chr === '.') {
115           key = key.slice(1);
116         }
117 
118         // Recurse
119         return Espresso.getObjectFor(key, object);
120 
121       // Attribute (`.`) subscript
122       } else if ((iattr < iarr || iarr === -1) && iattr > -1) {
123         object = getProperty(key.split('.', 1), object);
124         idx = iattr + 1;
125 
126         // Eat up the dot.
127         key = key.slice(iattr + 1);
128 
129         chr = key.charAt(0);
130         // Malformed property path
131         if (chr === "." || chr === '[') {
132           throw new Error(Espresso.format(pathError, chr, fullKey, idx + 1, '^'));
133         }
134 
135         // Recurse
136         return Espresso.getObjectFor(key, object);
137 
138       // Done!
139       } else if (key === '') {
140         return object;
141       }
142 
143       // Plain 'ol getObjectFor
144       return getProperty(key, object);
145     };
146   }()),
147 
148   /**
149     Checks whether the variable is defined *and* is not null.
150 
151     @param {Object} o The object to test if it's defined or not.
152     @returns {Boolean} True if the value is not null and not undefined.
153 
154     @example
155       var unbound;
156       undefined = 'bwahahaha!';
157       alert(Espresso.hasValue(unbound));
158       // -> false
159 
160       alert(Espresso.hasValue(undefined));
161       // -> true
162    */
163   hasValue: function (o) {
164     return o != null;
165   },
166 
167   /** @function
168     @desc
169 
170     Check to see if the object has function-like properties.
171     If it's callable, then it's a function or an object with
172     `call` and `apply` functions (which are assumed to work
173     how the same ones work on {@link Function.prototype}).
174 
175     @param {Object} obj The Object to check whether it is callable or not.
176     @returns {Boolean} True if the Object is callable, otherwise false.
177    */
178   isCallable: (function () {
179     var isFunction = '[object Function]',
180         isObject = '[object Object]',
181         toString = Object.prototype.toString,
182         nil = null;
183     return function (obj) {
184       return obj && (toString.call(obj) === isFunction ||
185              (obj.call != nil && toString.call(obj.call) === isFunction &&
186               obj.apply != nil && toString.call(obj.apply) === isFunction));
187     };
188   }()),
189 
190   /** @function
191     @desc
192     Convert an iterable object into an Array.
193 
194     @param {Object} iterable An iterable object with a length and indexing.
195     @returns {Array} The object passed in as an Array.
196    */
197   A: (function () {
198     var slice = Array.prototype.slice;
199     return function (iterable) {
200       return slice.apply(iterable);
201     };
202   }()),
203 
204   /**
205     Defers execution until a later time (when the ready
206     queue is empty).
207 
208     @param {Function} lambda The function to call.
209     @param {Array} args The arguments to apply to the function.
210     @param {Object} that The object to apply as `this`.
211    */
212   defer: function (lambda, args, that) {
213     that = that || lambda;
214     setTimeout(function () {
215       lambda.apply(that, args);
216     }, 0);
217   }
