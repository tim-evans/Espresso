 32   /** @function
 33     @desc
 34     Date Formatting support (for use with `fmt`).
 35 
 36     The following flags are acceptable in a format string:
 37 
 38      - `a` The abbreviated weekday name ("Sun")
 39      - `A` The full weekday name ("Sunday")
 40      - `b` The abbreviated month name ("Jan")
 41      - `B` The full month name ("January")
 42      - `c` The preferred local date and time representation
 43      - `d` Day of the month (01..31)
 44      - `H` Hour of the day, 24-hour clock (00..23)
 45      - `I` Hour of the day, 12-hour clock (01..12)
 46      - `j` Day of the year (001..366)
 47      - `m` Month of the year (01..12)
 48      - `M` Minute of the hour (00..59)
 49      - `p` Meridian indicator ("AM" or "PM")
 50      - `S` Second of the minute (00..60)
 51      - `U` Week number of the current year, starting with the first Sunday as the first day of the first week (00..53)
 52      - `W` Week number of the current year, starting with the first Monday as the first day of the first week (00..53)
 53      - `w` Day of the week (Sunday is 0, 0..6)
 54      - `x` Preferred representation for the date alone, no time
 55      - `X` Preferred representation for the time alone, no date
 56      - `y` Year without a century (00..99)
 57      - `Y` Year with century
 58 
 59     For example:
 60 
 61         alert("Today is {:A, B d, Y}.".fmt(new Date()));
 62 
 63         alert("The time is: {:c}.".fmt(new Date()));
 64 
 65     Note: all times used with `fmt` are **NOT** in UTC.
 66 
 67     @param {String} spec The specifier to transform the date to a formatted string.
 68     @returns {String} The Date transformed into a string as specified.
 69    */
 70   __fmt__: (function () {
 71     var days = ["Sunday", "Monday", "Tuesday", "Wednesday", "Thursday", "Friday", "Saturday"],
 72         months = ["January", "February", "March", "April", "May", "June",
 73                   "July", "August", "September", "October", "November", "December"];
 74 
 75     return function (spec) {
 76       var result = [], i = 0;
 77 
 78       for (; i < spec.length; i += 1) {
 79         switch (spec[i]) {
 80         case 'a':
 81           result[result.length] = days[this.getDay()].slice(0, 3);
 82           break;
 83         case 'A':
 84           result[result.length] = days[this.getDay()];
 85           break;
 86         case 'b':
 87           result[result.length] = months[this.getMonth()].slice(0, 3);
 88           break;
 89         case 'B':
 90           result[result.length] = months[this.getMonth()];
 91           break;
 92         case 'c':
 93           result[result.length] = "{0:a b} {1:2} {0:H:M:S Y}".fmt(this, this.getDate());
 94           break;
 95         case 'd':
 96           result[result.length] = "{:02}".fmt(this.getDate());
 97           break;
 98         case 'H':
 99           result[result.length] = "{:02}".fmt(this.getHours());
100           break;
101         case 'I':
102           result[result.length] = "{:02}".fmt(this.getHours() % 12);
103           break;
104         case 'j':
105           result[result.length] = "{:03}".fmt(Math.ceil((this - new Date(this.getFullYear(), 0, 1)) / 86400000));
106           break;
107         case 'm':
108           result[result.length] = "{:02}".fmt(this.getMonth() + 1);
109           break;
110         case 'M':
111           result[result.length] = "{:02}".fmt(this.getMinutes());
112           break;
113         case 'p':
114           result[result.length] = this.getHours() > 11 ? "PM" : "AM";
115           break;
116         case 'S':
117           result[result.length] = "{:02}".fmt(this.getSeconds());
118           break;
119         case 'U':
120           // Monday as the first day of the week
121           var day = ((this.getDay() + 6) % 7) + 1;
122           result[result.length] = "{:02}".fmt(
123             Math.ceil((((this - new Date(this.getFullYear(), 0, 1)) / 86400000) + day) / 7) - 1);
124           break;
125         case 'w':
126           result[result.length] = this.getDay();
127           break;
128         case 'W':
129           result[result.length] = "{:02}".fmt(
130             Math.ceil((((this - new Date(this.getFullYear(), 0, 1)) / 86400000) + this.getDay() + 1) / 7) - 1);
131           break;
132         case 'x':
133           result[result.length] = "{:m/d/y}".fmt(this);
134           break;
135         case 'X':
136           result[result.length] = this.toLocaleTimeString();
137           break;
138         case 'y':
139           result[result.length] = "{:02}".fmt(this.getYear() % 100);
140           break;
141         case 'Y':
142           result[result.length] = this.getFullYear();
143           break;
144         default:
145           result[result.length] = spec[i];
146         }
147       }
148       return result.join('');
149     };
150   }())
151 }).into(Date.prototype);
