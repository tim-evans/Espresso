  3 /** @function
  4   @desc
  5   Mix in functionality to a pre-existing object.
  6 
  7   At the base level, `mix` will add the properties
  8   given on `mix` to the object passed in on `into`.
  9 
 10   Function decorators are told to alter the function
 11   on mixin time, rather than at decoration time.
 12   Function decorators do things like subscribe to
 13   events, tell `mix` to ignore it, ask to be sent
 14   the super class from the base object, and so on.
 15 
 16   Using the decorator interface to inject your own
 17   library's custom work in is fairly simple. The
 18   guidelines are that decorators **should** propagate
 19   through inheritance and mixins. Decorators should
 20   also not interfere with other decorator's behaviour.
 21 
 22   If you would like to mixin functionality to
 23   preexisting objects, use `mix` to do so, using the
 24   Object as the second parameter, like so:
 25 
 26       // Simple screenplay reader.
 27       var screenplay = {
 28         dialogue: function (speaker, dialogue) {
 29           alert("{}: {}".fmt(speaker, dialogue));
 30         },
 31 
 32         scene: function () {
 33           var args = Espresso.toArray(arguments);
 34           args.forEach(function (line) {
 35             this.dialogue.apply(this, line);
 36           }, this);
 37         }
 38       };
 39 
 40       // Add the Spanish Inquisition.
 41       mix({
 42         dialogue: function (original, speaker, dialogue) {
 43           original(speaker, dialogue);
 44           if (dialogue.indexOf("Spanish Inquisition") !== -1) {
 45             original("Cardinal Ximinez",
 46                      "Nobody Expects the Spanish Inquisition!");
 47           }
 48         }.refine()
 49       }).into(screenplay);
 50 
 51       screenplay.scene(
 52         ["Chapman",   "Trouble at the mill."],
 53         ["Cleveland", "Oh no- what kind of trouble?"],
 54         ["Chapman",   "One on't cross beams gone owt askew on treadle."],
 55         ["Cleveland", "Pardon?"],
 56         ["Chapman",   "One on't cross beams gone owt askew on treadle."],
 57         ["Cleveland", "I don't understand what you're saying."],
 58         ["Chapman",   "One of the cross beams gone out askew on the treadle."],
 59         ["Cleveland", "Well, what on earth does that mean?"],
 60         ["Chapman",   "I don't know- Mr. Wentworth just told me to come in here " +
 61                       "and say that there was trouble at the mill, that's all- " +
 62                       "I didn't expect a kind of Spanish Inquisition!"]
 63       );
 64 
 65   Using `mix`, it's possible to create whatever types
 66   of objects you want, without polluting it's namespace.
 67   Espresso uses `mix` internally as a shim for ECMAScript 5
 68   compatability and creating the core of your library.
 69 
 70   @param {...} mixins Objects to mixin to the template provided on into.
 71   @returns {Object} An object with `into` field, call into with the template
 72                     to apply the mixins on. That will return the template
 73                     with the mixins on it.
 74  */
 75 mix = function () {
 76   var mixins = arguments,
 77       i = 0, len = mixins ? mixins.length : 0;
 78 
 79   return {
 80     into: function (template) {
 81       var mixin, key, value,
 82           _, decorator;
 83 
 84       if (!Espresso.hasValue(template)) {
 85         throw new TypeError("Cannot mix into null or undefined values.");
 86       }
 87 
 88       for (; i < len; i += 1) {
 89         mixin = mixins[i];
 90         for (key in mixin) {
 91           value = mixin[key];
 92 
 93           _ = value && value._;
 94           if (Espresso.isCallable(value) && _) {
 95             for (decorator in _) {
 96               if (_.hasOwnProperty(decorator)) {
 97                 value = _[decorator](template, value, key);
 98               }
 99             }
100           }
101 
102           template[key] = value;
103         }
104 
105         // Take care of IE clobbering toString and valueOf
106         if (mixin && mixin.toString !== Object.prototype.toString) {
107           template.toString = mixin.toString;
108         } else if (mixin && mixin.valueOf !== Object.prototype.valueOf) {
109           template.valueOf = mixin.valueOf;
110         }
111       }
112       return template;
113     }
114   };
115 };
116 
117 // Apply it at the global scope
118 Espresso.global.mix = mix;
