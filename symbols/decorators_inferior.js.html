1 mix(/** @scope Espresso */{
  2 
  3   /**
  4     If the attribute being mixed in exists on the
  5     Object being mixed in, the object marked as
  6     inferior will **not** be mixed in. If the base
  7     object is inferior, it will be overriden.
  8 
  9     @param {Object} target The target to apply the decorator to.
 10     @param {Object|Function} [condition] If it returns `true`,
 11       the function is inferior. Otherwise, it isn't.
 12     @returns {Function} The reciever.
 13    */
 14   inferior: function (target, condition) {
 15     var isInferior = arguments.length === 2 ?
 16       (Espresso.isCallable(condition) ? condition() : condition) : true;
 17     if (!isInferior) { return target; }
 18 
 19     target._ = target._ || {};
 20     target.isInferior = true;
 21 
 22     /** @ignore */
 23     target._.inferior = function (template, value, key) {
 24       return (!template[key] || template[key].isInferior) ? value: template[key];
 25     };
 26 
 27     return target;
 28   }
 29 
 30 }).into(Espresso);
