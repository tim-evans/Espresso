1 /** @namespace
  2   Publish-Subscribe mixin that provides the basics of eventing.
  3 
  4   @example
  5     var sailor = mix(Espresso.PubSub, {
  6       name: "",
  7       ahoy: function (action, sailor) {
  8         alert("{0.name}: Ahoy, {1.name}!".fmt(this, sailor));
  9       }
 10     }).into({});
 11 
 12     var ship = mix(Espresso.PubSub, {
 13       sailors: [],
 14 
 15       add: function (sailor, sync) {
 16         this.sailors.push(sailor);
 17         alert("Added {name}".fmt(sailor));
 18         this.publish("add", sailor);
 19        this.subscribe("add", sailor.ahoy.bind(sailor), { synchronous: !!sync });
 20       }
 21     }).into({});
 22 
 23     var ahab = mix(sailor, { name: "Captain Ahab" }).into({}),
 24         daveyJones = mix(sailor, { name: "Davey Jones" }).into({}),
 25         flapjack = mix(sailor, { name: "Flapjack" }).into({});
 26 
 27     ship.add(ahab, true);
 28     ship.add(daveyJones);
 29     ship.add(flapjack);
 30  */
 31 /*global mix Espresso */

 33 Espresso.PubSub = /** @lends Espresso.PubSub# */{
 34 
 35   /** @private */
 36   _subscriptions: null,
 37 
 38   /**
 39    * Subscribe to an event.
 40    *
 41    * @param {Object} event The event to subscribe to.
 42    * @param {Function} handler The handler to call when the event is published.
 43    * @param {Object} [options] Optional parameters.
 44    *   @param {Boolean} [options.synchronous] Whether the handler should be called synchronously or not. Defaults to asynchronous calls.
 45    * @returns {Object} The reciever.
 46    */
 47   subscribe: function (event, handler, options) {
 48     if (!Espresso.isCallable(handler)) {
 49       throw new TypeError("{} is not callable.".fmt(handler));
 50     }
 51 
 52     var subscriptions = this._subscriptions || {};
 53     if (!subscriptions[event]) {
 54       subscriptions[event] = [];
 55     }
 56 
 57     subscriptions[event].push(mix(options, {
 58       subscriber: handler
 59     }).into({}));
 60 
 61     this._subscriptions = subscriptions;
 62     return this;
 63   },
 64 
 65   /**
 66    * Unsubscribe from an event.
 67    *
 68    * @param {Object} event The event to subscribe to.
 69    * @param {Function} handler The handler to call when the event is published.
 70    * @returns {Object} The reciever.
 71    */
 72   unsubscribe: function (event, handler) {
 73     var subscriptions = this._subscriptions, handlers, i, len;
 74     if (subscriptions && subscriptions[event]) {
 75       handlers = subscriptions[event];
 76       for (i = 0, len = handlers.length; i < len; i += 1) {
 77         if (handlers[i].subscriber === handler) {
 78           subscriptions[event].remove(i);
 79           break;
 80         }
 81       }
 82     }
 83     return this;
 84   },
 85 
 86   /**
 87    * Gets called when an event has no subscribers to it.
 88    * Override to handle the case when nothing is published.
 89    *
 90    * @param {Object} event The event that was ignored.
 91    */
 92   unpublishedEvent: function (event) {},
 93 
 94   /**
 95    * Publish an event, passing all arguments along to the subscribed functions.
 96    *
 97    * @param {Object} event The event to publish.
 98    * @returns {Object} The reciever.
 99    */
100   publish: function (event) {
101     var subscriptions = this._subscriptions,
102         args = arguments, subscriber, published = false;
103     if (subscriptions && subscriptions[event]) {
104       subscriptions[event].forEach(function (subscription) {
105         subscriber = subscription.subscriber;
106         if (subscription.synchronous) {
107           Espresso.Scheduler.invoke(subscriber, args, this);
108         } else {
109           Espresso.Scheduler.defer(subscriber, args, this);
110         }
111         published = true;
112       }, this);
113     }
114     if (!published && Espresso.isCallable(this.unpublishedEvent)) {
115       Espresso.Scheduler.invoke(this.unpublishedEvent, arguments, this);
116     }
117     return this;
118   }
119 };

121 /** @namespace
122 
123   The scheduler is a mechanism to call functions in an
124   abstract fashion. The built-in implementation is
125   simplistic and may not meet the needs of your library.
126 
127   It's here so you may swap out functionality to suit your
128   needs without mucking with moving parts within Espresso.
129   You may interpret the functions as you see fit. Just mind
130   that mucking with their implementation *will* change how
131   notifications are delivered for the {@link Espresso.PubSub}
132   and {@link Espresso.KVO} mixins.
133  */
134 Espresso.Scheduler = {
135 
136   /**
137     Defers execution until a later time (when the ready
138     queue is empty).
139 
140     @param {Function} lambda The function to call.
141     @param {Array} args The arguments to apply to the function.
142     @param {Object} that The object to apply as `this`.
143    */
144   defer: function (lambda, args, that) {
145     that = that || lambda;
146     setTimeout(function () {
147       lambda.apply(that, args);
148     }, 0);
149   },
150 
151   /**
152     Invokes a function immediately.
153 
154     @param {Function} lambda The function to call.
155     @param {Array} args The arguments to apply to the function.
156     @param {Object} that The object to apply as `this`.
157    */
158   invoke: function (lambda, args, that) {
159     that = that || lambda;
160     lambda.apply(that, args);
161   }
162 };
