 27   /**
 28     Provides a mechanism to alias a function with
 29     other names on the object.
 30 
 31     Any arguments passed in will be used as aliases
 32     for the function. Each of these aliases will be
 33     references to the original, meaning that all of
 34     them will be indistinguishable and if one is
 35     altered in place, then all will be.
 36 
 37     @param {...} aliases The aliases this function has.
 38     @returns {Function} The reciever.
 39    */
 40   alias: function () {
 41     this._ = this._ || {};
 42 
 43     var aliases = Array.from(arguments),
 44         idx = aliases.length, mixin;
 45 
 46     /** @ignore */
 47     this._.alias = function (template, value, key) {
 48       delete value._.alias; // Remove this to prevent recursion.
 49       while (idx--) {
 50         mixin = {};
 51         mixin[aliases[idx]] = value;
 52         mix(mixin).into(template);
 53       }
 54       return value;
 55     };
 56 
 57     return this;
 58   },

 60   /**
 61     Around provides `super` functionality to a function.
 62     When the decorated function is called, it will have it's
 63     first argument bound to the function this one will override.
 64 
 65     If this function will not override anything, then the first
 66     argument will be an empty function that returns nothing.
 67 
 68     The `super` function will always be in the current scope of
 69     the function being called at the moment. Since the scope is
 70     maintained for you, you **must** make sure that scope that the
 71     first in the chain is what you want all down the chain. This
 72     is the typical behaviour of `super` in other languages;
 73     therefore it is done for you, and forces you into that
 74     situation.
 75 
 76     NOTE: If you try to rebind the property using
 77           {@link Function#bind}, it will _not_ work.
 78 
 79     @returns {Function} The reciever.
 80    */
 81   around: function () {
 82     this._ = this._ || {};
 83 
 84     var empty = function () {};
 85 
 86     /** @ignore */
 87     this._.around = function (template, value, key) {
 88       var base = template[key] || empty;
 89       if (!Espresso.isCallable(base)) {
 90         return value;
 91       }
 92 
 93       return function () {
 94         return value.apply(this, [base.bind(this)].concat(Array.from(arguments)));
 95       };
 96     };
 97     return this;

100   /**
101     Notifies the function when a property did change.
102     The notification will be delivered asynchronously
103     to the function.
104 
105     `on` is intended to emphasize the publish/subscribe
106     pattern, where messages are delivered asynchronously
107     to the method, with the intention that the method
108     is foreign code that absolutely cannot disturb the
109     execution of the library.
110 
111     All `on` does is subscribes the function to the
112     property paths given using asynchronous delivery.
113 
114     @param {...} subscriptions The events to be
115       notified when anything gets published to them.
116     @returns {Function} The reciever.
117     @see Function#observes
118    */
119   on: function () {
120     this._ = this._ || {};
121 
122     var pubsub = Array.from(arguments);
123 
124     /** @ignore */
125     this._.on = function (template, value, key) {
126       var i = 0, len = pubsub.length, object, property, iProperty;
127       for (i = 0; i < len; i += 1) {
128         property = pubsub[i];
129         object = template;
130 
131         if (property.indexOf('.') !== -1) {
132           iProperty = property.lastIndexOf('.');
133           object = Espresso.getObjectFor(property.slice(0, iProperty));
134           property = property.slice(iProperty + 1);
135         }
136 
137         if (object && object.subscribe && object.publish) {
138           object.subscribe(property, value);
139         }
140       }
141       return value;
142     };
143     return this;
144   },

146   /**
147     Notifies the function when a property did change.
148     The notification will be delivered synchronously
149     to the function.
150 
151     `observes` is intended to emphasize the observer
152     pattern, where messages are delivered synchronously
153     to the method. This is extremely useful for taking
154     action when an internal event needs to be propagated
155     to many internal sources.
156 
157     Try not to expose `observes` as a public API method,
158     as it will potentially crash the library's code if
159     _any_ error is thrown.
160 
161     All `observes` does is subscribes the function to the
162     property paths given using synchronous delivery.
163 
164     Here's a simple clock using observers to propagate
165     changes from appropriate levels outward.
166 
167         var Clock = Espresso.Class.extend({
168           time: null,
169           timer: null,
170 
171           init: function () {
172             this.set('timer', setInterval(this.tick.bind(this), 1000));
173             this.tick();
174           },
175 
176           tick: function () {
177             this.set('time', new Date(Date.now()));
178           },
179 
180           timeDidChange: function () {
181             alert("{:c}".fmt(this.get('time')));
182           }.observes('time')
183         });
184 
185         var clock = new Clock();
186         setTimeout(function () {
187           clearInterval(clock.get('timer'))
188         }, 5000);
189 
190     @param {...} observers The property paths to be
191       notified when anything gets published to them.
192     @returns {Function} The reciever.
193     @see Function#on
194    */
195   observes: function () {
196     this._ = this._ || {};
197 
198     var pubsub = Array.from(arguments);
199 
200     /** @ignore */
201     this._.observes = function (template, value, key) {
202       var i = 0, len = pubsub.length, object, property, iProperty;
203       for (i = 0; i < len; i += 1) {
204         property = pubsub[i];
205         object = template;
206 
207         if (property.indexOf('.') !== -1) {
208           iProperty = property.lastIndexOf('.');
209           object = Espresso.getObjectFor(property.slice(0, iProperty));
210           property = property.slice(iProperty + 1);
211         }
212 
213         if (object && object.subscribe && object.publish) {
214           object.subscribe(property, value, { synchronous: true });
215         }
216       }
217       return value;
218     };
219     return this;

300   /**
301     Bind the value of `this` on a function before hand,
302     with any extra arguments being passed in as initial
303     arguments.
304 
305     This implementation conforms to the ECMAScript 5
306     standard.
307 
308         var Person = Espresso.Template.extend({
309           name: 'nil',
310           greet: function (greeting) {
311             alert(greeting.fmt(this.name));
312           }
313         });
314 
315         var wash = Person.extend({
316           name: 'Hoban Washburne'
317         });
318 
319         var mal = Person.extend({
320           name: 'Malcolm Reynolds'
321         });
322 
323         mal.greet("Hello, {}!");
324         // -> "Hello, Malcolm Reynolds!"
325 
326         var greet = mal.greet.bind(wash);
327         greet("Howdy, {}!");
328         // -> "Howdy, Hoban Washburne!"
329 
330     @param {Object} thisArg The value to bind `this` to on the function.
331     @returns {Function} The function passed in, wrapped to ensure `this`
332       is the correct scope.
333    */
334   bind: function (self) {
335     var Target, A;
336 
337     // 1. Let Target be the this value.
338     Target = this;
339 
340     // 2. If IsCallable(Target) is false, throw a TypeError exception
341     if (!Espresso.isCallable(Target)) {
342       throw new TypeError("The Target is not callable.");
343     }
344 
345     // 3. Let A be a new (possibly empty) internal list of
346     //    all argument values provided after self
347     //    (arg1, arg2, etc), in order
348     A = Array.from(arguments).slice(1);
349 
350     var bound = function () {
351       
352       if (this instanceof bound) {
353         // 15.3.4.5.2 [[Construct]]
354         // When the [[Construct]] internal method of a function object, F,
355         // that was created using the bind function is called with a list of
356         // arguments ExtraArgs, the following steps are taken:
357         var Type = function () {}, that;
358         Type.prototype = Target.prototype;
359         that = new Type();
360 
361         Target.apply(self, A.concat(Array.from(arguments)));
362         return that;
363       } else {
364         // 15.3.4.5.1 [[Call]]
365         // When the [[Call]] internal method of a function object, F,
366         // which was created using the bind function is called with a this
367         // value and a list of arguments ExtraArgs, the following steps are taken:
368         // 1. Let boundArgs be the value of F's [[BoundArgs]] internal property.
369         // 2. Let boundThis be the value of F's [[BoundThis]] internal property.
370         // 3. Let target be the value of F's [[TargetFunction]] internal property.
371         return Target.apply(self, A.concat(Array.from(arguments)));
372       }
373     };
374     return bound;
375   }.inferior(),
