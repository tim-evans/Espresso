  3 /** @function
  4   @desc
  5   Mix in functionality to a pre-existing object.
  6 
  7   At the base level, `mix` will add the properties
  8   given on `mix` to the object passed in on `into`.
  9 
 10   Function decorators are told to alter the function
 11   on mixin time, rather than at decoration time.
 12   Function decorators do things like subscribe to
 13   events, tell `mix` to ignore it, ask to be sent
 14   the super class from the base object, and so on.
 15 
 16   Using the decorator interface to inject your own
 17   library's custom work in is fairly simple. The
 18   guidelines are that decorators **should** propagate
 19   through inheritance and mixins. Decorators should
 20   also not interfere with other decorator's behaviour.
 21 
 22   If you would like to mixin functionality to
 23   preexisting objects, use `mix` to do so, using the
 24   Object as the second parameter, like so:
 25 
 26       mix({
 27         gsub: function (find, replace) {
 28           if (/string/i.test(Object.prototype.toString.call(find))) {
 29             find = new RegExp(find, 'g');
 30           }
 31           return this.replace(find, replace);
 32         }
 33       }).into(String.prototype);
 34 
 35       var song = "I swiped your cat / And I stole your cathodes"
 36       alert(song.gsub('cat', 'banjo'));
 37 
 38       alert(song.gsub(/\bcat\b/, 'banjo'));
 39 
 40   Using `mix`, it's possible to create whatever types
 41   of objects you want, without polluting it's namespace.
 42   Espresso uses `mix` internally as a shim for ECMAScript 5
 43   compatability and creating the base objects
 44   {@link Espresso.Template} and {@link Espresso.Class}.
 45 
 46   @param {...} mixins Objects to mixin to the template provided on into.
 47   @returns {Object} An object with "into" field, call into with the template
 48                     to apply the mixins on. That will return the template
 49                     with the mixins on it.
 50  */
 51 mix = function () {
 52   var mixins = arguments,
 53       i = 0, len = mixins ? mixins.length : 0;
 54 
 55   return {
 56     into: function (template) {
 57       var mixin, key, value,
 58           _, decorator;
 59 
 60       for (; i < len; i += 1) {
 61         mixin = mixins[i];
 62         for (key in mixin) {
 63           value = mixin[key];
 64 
 65           _ = value && value._;
 66           if (Espresso.isCallable(value)) {
 67             for (decorator in _) {
 68               if (_.hasOwnProperty(decorator)) {
 69                 value = _[decorator](template, value, key);
 70               }
 71             }
 72           }
 73 
 74           template[key] = value;
 75         }
 76 
 77         // Take care of IE clobbering toString and valueOf
 78         if (mixin && mixin.toString !== Object.prototype.toString) {
 79           template.toString = mixin.toString;
 80         } else if (mixin && mixin.valueOf !== Object.prototype.valueOf) {
 81           template.valueOf = mixin.valueOf;
 82         }
 83       }
 84       return template;
 85     }
 86   };
 87 };
 88 
 89 // Apply it at the global scope
 90 Espresso.global.mix = mix;
