 54   /** @function
 55     @desc
 56     Date Formatting support.
 57 
 58     The following flags are acceptable in a format string:
 59 
 60      - `a` The abbreviated weekday name ("Sun")
 61      - `A` The full weekday name ("Sunday")
 62      - `b` The abbreviated month name ("Jan")
 63      - `B` The full month name ("January")
 64      - `c` The preferred local date and time representation
 65      - `d` Day of the month (01..31)
 66      - `H` Hour of the day, 24-hour clock (00..23)
 67      - `I` Hour of the day, 12-hour clock (01..12)
 68      - `j` Day of the year (001..366)
 69      - `m` Month of the year (01..12)
 70      - `M` Minute of the hour (00..59)
 71      - `p` Meridian indicator ("AM" or "PM")
 72      - `S` Second of the minute (00..60)
 73      - `U` Week number of the current year, starting with the first Sunday as the first day of the first week (00..53)
 74      - `W` Week number of the current year, starting with the first Monday as the first day of the first week (00..53)
 75      - `w` Day of the week (Sunday is 0, 0..6)
 76      - `x` Preferred representation for the date alone, no time
 77      - `X` Preferred representation for the time alone, no date
 78      - `y` Year without a century (00..99)
 79      - `Y` Year with century
 80      - `Z` Timezone name or abbreviation (EST)
 81 
 82     For example:
 83 
 84         alert("Today is {:A, B d, Y}.".fmt(new Date()));
 85 
 86         alert("The time is: {:c}.".fmt(new Date()));
 87

 89     @returns {String} The Date transformed into a string as specified.
 90    */

 92     var days = ["Sunday", "Monday", "Tuesday", "Wednesday", "Thursday", "Friday", "Saturday"],
 93         months = ["January", "February", "March", "April", "May", "June",
 94                   "July", "August", "September", "October", "November", "December"];
 95 
 96     return function (spec) {
 97       var result = [], i = 0;
 98 
 99       for (; i < spec.length; i += 1) {
100         switch (spec[i]) {
101         case 'a':
102           result[result.length] = days[this.get('day')].slice(0, 3);
103           break;
104         case 'A':
105           result[result.length] = days[this.get('day')];
106           break;
107         case 'b':
108           result[result.length] = months[this.get('month')].slice(0, 3);
109           break;
110         case 'B':
111           result[result.length] = months[this.get('month')];
112           break;
113         case 'c':
114           result[result.length] = "{0:a b} {1:2} {0:H:M:S Y}".fmt(this, this.get('date'));
115           break;
116         case 'd':
117           result[result.length] = "{:02}".fmt(this.get('date'));
118           break;
119         case 'e':
120           result[result.length] = "{: 2}".fmt(this.get('date'));
121           break;
122         case 'f':
123           result[result.length] = "{:03}".fmt(this.get('milliseconds'));
124           break;
125         case 'H':
126           result[result.length] = "{:02}".fmt(this.get('hours'));
127           break;
128         case 'I':
129           result[result.length] = "{:02}".fmt(this.get('hours') % 12);
130           break;
131         case 'j':
132           result[result.length] = "{:03}".fmt(Math.ceil((this - new Date(this.get('fullYear'), 0, 1)) / 86400000));
133           break;
134         case 'm':
135           result[result.length] = "{:02}".fmt(this.get('month') + 1);
136           break;
137         case 'M':
138           result[result.length] = "{:02}".fmt(this.get('minutes'));
139           break;
140         case 'p':
141           result[result.length] = this.get('hours') > 11 ? "PM" : "AM";
142           break;
143         case 'S':
144           result[result.length] = "{:02}".fmt(this.get('seconds'));
145           break;
146         case 'U':
147           // Monday as the first day of the week
148           var day = ((this.get('day') + 6) % 7) + 1;
149           result[result.length] = "{:02}".fmt(
150             Math.ceil((((this - new Date(this.get('fullYear'), 0, 1)) / 86400000) + day) / 7) - 1);
151           break;
152         case 'w':
153           result[result.length] = this.get('day');
154           break;
155         case 'W':
156           result[result.length] = "{:02}".fmt(
157             Math.ceil((((this - new Date(this.get('fullYear'), 0, 1)) / 86400000) + this.get('day') + 1) / 7) - 1);
158           break;
159         case 'x':
160           result[result.length] = "{:m/d/y}".fmt(this);
161           break;
162         case 'X':
163           result[result.length] = this.toLocaleTimeString();
164           break;
165         case 'y':
166           result[result.length] = "{:02}".fmt(this.getYear() % 100);
167           break;
168         case 'Y':
169           result[result.length] = this.get('fullYear');
170           break;
171         case 'Z':
172           result[result.length] = this.get('timezoneOffset');
173           break;
174         default:
175           result[result.length] = spec[i];
176         }
177       }
178       return result.join('');
179     };
