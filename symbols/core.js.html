  3 /** @namespace
  4 
  5   Espresso is a JavaScript library to be used as a
  6   foundation library to create JavaScript libraries.
  7   It also acts as a partial shim to ECMAScript 5,
  8   falling back to native browser support when available.
  9 
 10   Espresso's goal is to provide a small library that
 11   provides the basics that provide the power to
 12   developers to produce sophisticated JavaScript libraries
 13   that have clear, concise, and readable code, as well as
 14   powerful consumer-facing APIs.
 15 
 16   What does this mean? Less code and robust APIs!
 17 
 18   This library provides the Publish-Subscribe pattern,
 19   Key-Value Observing (a la Cocoa), and Ruby-like mixins.
 20  */
 21 Espresso = {
 22 
 23   /**
 24     The version string.
 25     @type String
 26    */
 27   VERSION: '0.5.2',
 28 
 29   /**
 30     The global variable.
 31 
 32     Used to be independant of what the global `this` is,
 33     whether it's `window` or `document` in a browser or
 34     `global` in NodeJS.
 35 
 36     @type Object
 37    */
 38   global: this,
 39 
 40   /** @function
 41     @desc
 42 
 43     Lookup a variable's value given its Object notation.
 44     This requires absolute queries to the Object, only using
 45     the `.` notation.
 46 
 47     The most effort that is performed on behalf of the
 48     lookup when it fails is when it's an array AND it's the
 49     only element in the array, THEN it will unpack the element
 50     and make that the argument.
 51 
 52     This does not mean that absolute notation does not
 53     work in these cases; it just means that it's optional.
 54 
 55     This prevents unnecessary indexing by the user,
 56     expecially in the case of the arguments Array.
 57 
 58     @example
 59       // Properties on the global scope need to be there-
 60       // local scoped variables will not be found!
 61       Espresso.global.arthur = Espresso.Template.extend({
 62         name: 'Arthur Dent',
 63         species: 'Human',
 64         description: 'Mostly Harmless'
 65       });
 66 
 67       alert(Espresso.getObjectFor("arthur.name"));
 68       // => 'Arthur Dent'
 69 
 70     @example
 71       alert(Espresso.getObjectFor("lang.pr._coffee", {
 72         lang: {
 73           en: { _coffee: "coffee" },
 74           pr: { _coffee: "cafe" }
 75         }
 76       }));
 77       // -> "cafe"
 78 
 79     @param {String} key The key to get on the target.
 80     @param {Object} [object] The target object to get a value from.
 81     @returns {Object} The referenced value in the args passed in.
 82    */
 83   getObjectFor: (function () {
 84     /** @ignore */
 85     var getProperty = function (property, obj) {
 86       if (property in obj) {
 87         obj = obj[property];
 88       } else {
 89         // Try to be helpful-
 90         //  1) If the property doesn't exist on the object,
 91         //  2) The object is an Array
 92         //  3) The Array has only one element in it.
 93         // Unpack the element and try the lookup again.
 94         if (Array.isArray(obj) && obj.length === 1) {
 95           obj = obj[0];
 96         }
 97         if (obj && property in obj) { // prevent the apocalypse
 98           obj = obj[property];
 99         } else {
100           obj = undefined;
101         }
102       }
103       return obj;
104     };
105 
106     return function (key, object) {
107       // Array / Attribute subscript
108       var iattr = key.indexOf('.');
109 
110       // Use global scope as default
111       object = (arguments.length === 1) ? this.global: object;
112 
113       // Nothing to look up on undefined or null objects.
114       if (!Espresso.hasValue(object)) {
115         return object;
116       }
117 
118       if (iattr > -1) {
119         object = getProperty(key.split('.', 1), object);
120 
121         // Eat up the dot.
122         key = key.slice(key.indexOf('.') + 1);
123 
124         // Recurse
125         return Espresso.getObjectFor(key, object);
126 
127         // Done!
128       } else if (key === '') {
129         return object;
130       }
131 
132       // Plain 'ol getObjectFor
133       return getProperty(key, object);
134     };
135   }()),
136 
137   /**
138     Checks whether the variable is defined *and* is not null.
139 
140     @param {Object} o The object to test if it's defined or not.
141     @returns {Boolean} True if the value is not null and not undefined.
142 
143     @example
144       var unbound; // This variable is very lonely (and very much undefined)
145       undefined = 'all your base belong to us'; // Yes, you can rename undefined, but...
146       alert(Espresso.hasValue(unbound));
147       // -> false
148 
149       alert(Espresso.hasValue(undefined));
150       // -> true
151    */
152   hasValue: function (o) {
153     return (typeof o !== "undefined" && o !== null);
154   },
155 
156   /** @function
157     @desc
158 
159     ECMAScript compliant isCallable.
160 
161     > The abstract operation IsCallable determines if its argument,
162     > which must be an ECMAScript language value, is a callable function
163     > Object if it's an Object that hass a function called `call`.
164 
165     This allows overriding `call` on an object, effectively making it
166     a callable object.
167 
168     The one addition is ensuring that the method is also applicable,
169     (having the `apply` being callable too).
170 
171     @param {Object} obj The Object to check whether it is callable or not.
172     @returns {Boolean} True if the Object is callable, otherwise false.
173    */
174   isCallable: (function () {
175     var callable = /[Function|Object]/,
176         toString = Object.prototype.toString;
177     return function (obj) {
178       return !!(obj && callable.test(toString.call(obj)) &&
179                 Espresso.hasValue(obj.call) &&
180                 callable.test(toString.call(obj.call)) &&
181                 Espresso.hasValue(obj.apply) &&
182                 callable.test(toString.call(obj.apply)));
183     };
184   }())
185 };
