1 /** @namespace
  2   Publish-Subscribe is a design pattern that allows
  3   event broadcasting to subscribed handlers. When an
  4   event is published to a node, the event is broadcasted
  5   to all subscribed handlers.
  6 
  7   Events can be filtered at runtime, which can tell
  8   PubSub whether or not it should publish events to
  9   that handler, and they can be sent either synchronously
 10   or asynchronously (the default is asynchronous).
 11 
 12   @example
 13     var ship = mix(Espresso.Subscribable, {
 14       sailors: [],
 15 
 16       add: function (sailor, sync) {
 17         this.sailors.push(sailor);
 18         alert("Added {name}".fmt(sailor));
 19         this.publish("add", sailor);
 20        this.subscribe("add", sailor.ahoy.bind(sailor), { synchronous: !!sync });
 21       }
 22     }).into({});
 23 
 24     var ahab = mix(sailor, { name: "Captain Ahab" }).into({}),
 25         daveyJones = mix(sailor, { name: "Davey Jones" }).into({}),
 26         flapjack = mix(sailor, { name: "Flapjack" }).into({});
 27 
 28     ship.add(ahab, true);
 29     ship.add(daveyJones);
 30     ship.add(flapjack);
 31  */
 32 /*global mix Espresso */
 33 
 34 Espresso.Subscribable = /** @lends Espresso.Subscribable# */{
 35 
 36   /** @private */
 37   _subscriptions: null,
 38 
 39   /**
 40     Subscribe to an event.
 41 
 42     @param {Object} event The event to subscribe to.
 43     @param {Function} handler The handler to call when the event is published.
 44     @param {Object} [options] Optional parameters.
 45       @param {Boolean} [options.synchronous] Whether the handler should be called synchronously or not. Defaults to asynchronous calls.
 46       @param {Function} [options.condition] A mechanism to refine whether a specific event is wanted. Return true if you would like the event, and false if you don't.
 47     @returns {Object} The reciever.
 48    */
 49   subscribe: function (event, handler, options) {
 50     if (!Espresso.isCallable(handler)) {
 51       throw new TypeError("{} is not callable.".fmt(handler));
 52     }
 53 
 54     var subscriptions = this._subscriptions || {};
 55     if (!subscriptions[event]) {
 56       subscriptions[event] = [];
 57     }
 58 
 59     if (options && options.condition && !Espresso.isCallable(options.condition)) {
 60       delete options.condition;
 61     }
 62     mix({ condition: function () { return true; }.inferior() }).into(options);
 63 
 64     subscriptions[event].push(mix(options, {
 65       subscriber: handler
 66     }).into({}));
 67 
 68     this._subscriptions = subscriptions;
 69     return this;
 70   },
 71 
 72   /**
 73     Unsubscribe from an event.
 74 
 75     @param {Object} event The event to subscribe to.
 76     @param {Function} handler The handler to call when the event is published.
 77     @returns {Object} The reciever.
 78    */
 79   unsubscribe: function (event, handler) {
 80     var subscriptions = this._subscriptions, handlers, i, len;
 81     if (subscriptions && subscriptions[event]) {
 82       handlers = subscriptions[event];
 83       for (i = 0, len = handlers.length; i < len; i += 1) {
 84         if (handlers[i].subscriber === handler) {
 85           subscriptions[event].splice(i, 1);
 86           break;
 87         }
 88       }
 89     }
 90     return this;
 91   },
 92 
 93   /**
 94     Gets called when an event has no subscribers to it.
 95 
 96     Override to handle the case when nothing is published.
 97     (There are no subscribers for an event.)
 98 
 99     Any parameters passed to the event are also passed into
100     the function. All unpublished events are `invoke`d rather
101     than `defer`red.
102 
103     @param {Object} event The event that was ignored.
104     @returns {void}
105    */
106   unpublishedEvent: function (event) {},
107 
108   /**
109     Publish an event, passing all arguments along to the subscribed functions.
110 
111     @param {Object} event The event to publish.
112     @returns {Object} The reciever.
113    */
114   publish: function (event) {
115     var subscriptions = this._subscriptions,
116         args = arguments, subscriber, published = false;
117     if (subscriptions && subscriptions[event]) {
118       subscriptions[event].forEach(function (subscription) {
119         if (Espresso.Scheduler.invoke(subscription.condition, args, this)) {
120           subscriber = subscription.subscriber;
121           if (subscription.synchronous) {
122             Espresso.Scheduler.invoke(subscriber, args, this);
123           } else {
124             Espresso.Scheduler.defer(subscriber, args, this);
125           }
126           published = true;
127         }
128       }, this);
129     }
130     if (!published && Espresso.isCallable(this.unpublishedEvent)) {
131       Espresso.Scheduler.invoke(this.unpublishedEvent, arguments, this);
132     }
133     return this;
134   }
135 };

137 /** @namespace
138 
139   The scheduler is a mechanism to call functions in an
140   abstract fashion. The built-in implementation is
141   simplistic and may not meet the needs of your library.
142 
143   It's here so you may swap out functionality to suit your
144   needs without mucking with moving parts within Espresso.
145   You may interpret the functions as you see fit. Just mind
146   that mucking with their implementation *will* change how
147   notifications are delivered for the {@link Espresso.Subscribable}
148   and {@link Espresso.Observable} mixins.
149  */
150 Espresso.Scheduler = {
151 
152   /**
153     Defers execution until a later time (when the ready
154     queue is empty).
155 
156     @param {Function} lambda The function to call.
157     @param {Array} args The arguments to apply to the function.
158     @param {Object} that The object to apply as `this`.
159    */
160   defer: function (lambda, args, that) {
161     that = that || lambda;
162     setTimeout(function () {
163       lambda.apply(that, args);
164     }, 0);
165   },
166 
167   /**
168     Invokes a function immediately.
169 
170     @param {Function} lambda The function to call.
171     @param {Array} args The arguments to apply to the function.
172     @param {Object} that The object to apply as `this`.
173    */
174   invoke: function (lambda, args, that) {
175     that = that || lambda;
176     return lambda.apply(that, args);
177   }
178 };
