  5   /**
  6     If the attribute being mixed in exists on the
  7     Object being mixed in, the function marked as
  8     inferior will **not** be mixed in.
  9 
 10     Also, if the base function is inferior, it
 11     will be overriden.
 12 
 13     @returns {Function} The reciever.
 14    */
 15   inferior: function () {
 16     this._ = this._ || {};
 17     this.isInferior = true;
 18 
 19     /** @ignore */
 20     this._.inferior = function (template, value, key) {
 21       return (!template[key] || template[key].isInferior) ? value: template[key];
 22     };
 23 
 24     return this;
 25   },
 26 
 27   /**
 28     Provides a mechanism to alias a function with
 29     other names on the object.
 30 
 31     Any arguments passed in will be used as aliases
 32     for the function. Each of these aliases will be
 33     references to the original, meaning that all of
 34     them will be indistinguishable and if one is
 35     altered in place, then all will be.
 36 
 37     @param {...} aliases The aliases this function has.
 38     @returns {Function} The reciever.
 39    */
 40   alias: function () {
 41     this._ = this._ || {};
 42 
 43     var aliases = Array.from(arguments),
 44         idx = aliases.length, mixin;
 45 
 46     /** @ignore */
 47     this._.alias = function (template, value, key) {
 48       delete value._.alias; // Remove this to prevent recursion.
 49       while (idx--) {
 50         mixin = {};
 51         mixin[aliases[idx]] = value;
 52         mix(mixin).into(template);
 53       }
 54       return value;
 55     };
 56 
 57     return this;
 58   },
 59 
 60   /**
 61     Around provides `super` functionality to a function.
 62     When the decorated function is called, it will have it's
 63     first argument bound to the function this one will override.
 64 
 65     If this function will not override anything, then the first
 66     argument will be an empty function that returns nothing.
 67 
 68     The `super` function will always be in the current scope of
 69     the function being called at the moment. Since the scope is
 70     maintained for you, you **must** make sure that scope that the
 71     first in the chain is what you want all down the chain. This
 72     is the typical behaviour of `super` in other languages;
 73     therefore it is done for you, and forces you into that
 74     situation.
 75 
 76     NOTE: If you try to rebind the property using
 77           {@link Function#bind}, it will _not_ work.
 78 
 79     @returns {Function} The reciever.
 80    */
 81   around: function () {
 82     this._ = this._ || {};
 83 
 84     var empty = function () {};
 85 
 86     /** @ignore */
 87     this._.around = function (template, value, key) {
 88       var base = template[key] || empty;
 89       if (!Espresso.isCallable(base)) {
 90         return value;
 91       }
 92 
 93       return function () {
 94         return value.apply(this, [base.bind(this)].concat(Array.from(arguments)));
 95       };
 96     };
 97     return this;
 98   },
 99 
100   /**
101     Marks the function as a computed property.
102     You may now use the function for `get` and `set`.
103 
104     @param {...} dependentKeys The property paths to be
105       notified when anything gets published to them.
106     @returns {Function} The reciever.
107    */
108   property: function () {
109     this._ = this._ || {};
110 
111     this.isProperty = true;
112     this.dependentKeys = Array.prototype.slice.apply(arguments);
113 
114     /** @ignore */
115     this._.property = function (template, value, key) {
116       var i = 0, len = value.dependentKeys.length, object, property, iProperty,
117           /** @ignore */
118           notifier = function () {
119             this.set(key);
120           };
121 
122       for (i = 0; i < len; i += 1) {
123         property = value.dependentKeys[i];
124         object = template;
125 
126         if (property.indexOf('.') !== -1) {
127           iProperty = property.lastIndexOf('.');
128           object = Espresso.getObjectFor(property.slice(0, iProperty));
129           property = property.slice(iProperty + 1);
130         }
131 
132         if (object && object.subscribe && object.publish) {
133           object.subscribe(property, notifier, { synchronous: true });
134         }
135       }
136 
137       return value;
138     };
139 
140     return this;
141   },

178   /**
179     Bind the value of `this` on a function before hand,
180     with any extra arguments being passed in as initial
181     arguments.
182 
183     This implementation conforms to the ECMAScript 5
184     standard.
185 
186         var Person = Espresso.Template.extend({
187           name: 'nil',
188           greet: function (greeting) {
189             alert(greeting.fmt(this.name));
190           }
191         });
192 
193         var wash = Person.extend({
194           name: 'Hoban Washburne'
195         });
196 
197         var mal = Person.extend({
198           name: 'Malcolm Reynolds'
199         });
200 
201         mal.greet("Hello, {}!");
202         // -> "Hello, Malcolm Reynolds!"
203 
204         var greet = mal.greet.bind(wash);
205         greet("Howdy, {}!");
206         // -> "Howdy, Hoban Washburne!"
207 
208     @param {Object} thisArg The value to bind `this` to on the function.
209     @returns {Function} The function passed in, wrapped to ensure `this`
210       is the correct scope.
211    */
212   bind: function (self) {
213     var Target, A;
214 
215     // 1. Let Target be the this value.
216     Target = this;
217 
218     // 2. If IsCallable(Target) is false, throw a TypeError exception
219     if (!Espresso.isCallable(Target)) {
220       throw new TypeError("The Target is not callable.");
221     }
222 
223     // 3. Let A be a new (possibly empty) internal list of
224     //    all argument values provided after self
225     //    (arg1, arg2, etc), in order
226     A = Array.from(arguments).slice(1);
227 
228     var bound = function () {
229       
230       if (this instanceof bound) {
231         // 15.3.4.5.2 [[Construct]]
232         // When the [[Construct]] internal method of a function object, F,
233         // that was created using the bind function is called with a list of
234         // arguments ExtraArgs, the following steps are taken:
235         var Type = function () {}, that;
236         Type.prototype = Target.prototype;
237         that = new Type();
238 
239         Target.apply(self, A.concat(Array.from(arguments)));
240         return that;
241       } else {
242         // 15.3.4.5.1 [[Call]]
243         // When the [[Call]] internal method of a function object, F,
244         // which was created using the bind function is called with a this
245         // value and a list of arguments ExtraArgs, the following steps are taken:
246         // 1. Let boundArgs be the value of F's [[BoundArgs]] internal property.
247         // 2. Let boundThis be the value of F's [[BoundThis]] internal property.
248         // 3. Let target be the value of F's [[TargetFunction]] internal property.
249         return Target.apply(self, A.concat(Array.from(arguments)));
250       }
251     };
252     return bound;
253   }.inferior(),

255   /**
256    * Curry will add arguments to a function, returning the function as-is

258   curry: function () {
259     var Target, A;
260 
261     // 1. Let Target be the this value.
262     Target = this;
263 
264     // 2. If IsCallable(Target) is false, throw a TypeError exception
265     if (!Espresso.isCallable(Target)) {
266       throw new TypeError("The Target is not callable.");
267     }
268 
269     // 3. Let A be a new (possibly empty) internal list of
270     //    all argument values (arg1, arg2, etc), in order
271     A = Array.from(arguments);
272 
273     return function () {
274       return Target.apply(this, A.concat(Array.from(arguments)));
275     };
276   }
