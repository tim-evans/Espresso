1 /*globals Espresso */
  2 
  3 /** @class
  4 
  5   A KVO compliant Object Hash class.
  6  
  7   @extends Espresso.Enumerable
  8   @extends Espresso.KVO
  9   @extends Espresso.Template
 10  */
 11 Espresso.Hash = Espresso.Template.extend(Espresso.Enumerable, Espresso.KVO, /** @lends Espresso.Hash# */{
 12 
 13   /**
 14     Iterator over the Hash.
 15 
 16     @param {Function} lambda The callback to call for each element.
 17     @param {Object} [self] The Object to use as this when executing the callback.
 18     @returns {void}
 19    */
 20   forEach: function (lambda, self) {
 21     var k, v;
 22     for (k in this) {
 23       v = this.get(k);
 24       if (k.get(0) !== '_' && !Espresso.isCallable(v) && k !== "unknownProperty") {
 25         lambda.call(self, v, k, this);
 26       }
 27     }
 28   },
 29 
 30   /**
 31     Return all of the iterable keys on the hash.
 32 
 33     @returns {Array} A list of all of the iterable keys on the hash.
 34     @example
 35       var alphabet = Espresso.Hash.extend({
 36         a: 00, b: 01, c: 02, d: 03, e: 04, f: 05,
 37         g: 06, h: 07, i: 08, j: 09, k: 10, j: 11,
 38         k: 12, l: 13, m: 14, n: 15, o: 16, p: 17,
 39         q: 18, r: 19, s: 20, t: 21, u: 22, v: 23,
 40         w: 24, x: 25, y: 26, z: 27
 41       });
 42 
 43       alert(alphabet.keys());
 44    */
 45   keys: function () {
 46     return this.map(function (v, k) {
 47       return k;
 48     });
 49   },
 50 
 51   /**
 52     Return all iterable values on the hash.
 53 
 54     @returns {Array} A list of all iterable values on the hash.
 55     @example
 56       var days = Espresso.Hash.extend({
 57         1: 'Sunday',
 58         2: 'Monday',
 59         3: 'Tuesday',
 60         4: 'Wednesday',
 61         5: 'Thursday',
 62         6: 'Friday',
 63         7: 'Saturday'
 64       });
 65 
 66       alert(days.values());
 67    */
 68   values: function () {
 69     return this.map(function (v, k) {
 70       return v;
 71     });
 72   },
 73 
 74   /**
 75     Convert the Hash into an Array of tuples.
 76 
 77     @returns {Array[]} An array of tuples.
 78    */
 79   toArray: function () {
 80     return this.map(function (v, k) {
 81       return [k, v];
 82     });
 83   }
 84 });
