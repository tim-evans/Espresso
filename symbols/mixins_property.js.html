1 /** @namespace
  2   A mixin to apply to callable objects that
  3   want to be a computed property. This means
  4   that the property will act like a getter /
  5   setter, but with notifications via KVO.
  6  */
  7 Espresso.Property = /** @lends Espresso.Property# */{
  8 
  9   /**
 10     Walk like a duck.
 11     @type Boolean
 12     @default true
 13    */
 14   isProperty: true,
 15 
 16   /**
 17     Whether or not the property should be
 18     cached when it gets recalculated.
 19     @type Boolean
 20     @default false
 21    */
 22   isCacheable: false,
 23 
 24   /**
 25     Whether the property is volatile or not.
 26     Defaults to being a volatile property.
 27     @type Boolean
 28     @default false
 29    */
 30   isIdempotent: false,
 31 
 32   /**
 33     The keys that this property depends on.
 34     If any of these keys change, the property
 35     should be notified it did so.
 36     @type Array
 37    */
 38   dependentKeys: null,
 39 
 40   /**
 41     Marks the property as cacheable.
 42     @returns {Espresso.Property} The property.
 43    */
 44    cacheable: function () {
 45      this.isCacheable = true;
 46      return this;
 47    },
 48 
 49   /**
 50     Marks the property as idempotent.
 51     @returns {Espresso.Property} The property.
 52    */
 53    idempotent: function () {
 54      this.isIdempotent = true;
 55      return this;
 56    }
 57 };
