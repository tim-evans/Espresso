 44   /**

 53   /**
 54     Iterator over the Array.
 55 
 56     Implemented to be in conformance with ECMA-262 Edition 5,
 57     so you will use the native `forEach` where it exists.
 58 
 59     @param {Function} lambda The callback to call for each element.
 60     @param {Object} [self] The Object to use as this when executing the callback.
 61     @returns {void}
 62    */
 63   forEach: function (lambda, self) {
 64     var len, k;
 65 
 66     // 3. Let len be ToUint32(lenValue).
 67     len = this.get('size');
 68 
 69     // 4. If IsCallable(lambda) is false, throw a TypeError exception
 70     if (!Espresso.isCallable(lambda)) {
 71       throw new TypeError("{} is not callable.".fmt(lambda));
 72     }
 73 
 74     // 6. Let k be 0.
 75     k = 0;
 76 
 77     // 7. Repeat, while k < len
 78     while (k < len) {
 79       // c. If kPresent is true, then
 80       if (this.hasOwnProperty(k)) {
 81         //  i. Let kValue be the result of calling the [[Get]]
 82         //     internal method of O with argument Pk.
 83         // ii. Call the [[Call]] internal method of lambda
 84         //     with T as the this value and argument list
 85         //     containing kValue, k, and O.
 86         lambda.call(self, this.get(k), k, this);
 87       }
 88 
 89       // d. Increase k by 1.
 90       k += 1;
 91     }
 92 
 93     // 8. Return
 94   }.inferior(),

 96   /**
 97     Shim for Internet Explorer, which provides no `indexOf` for
 98     Array prototypes.
 99 
100     @param {Object} o The object to test.
101     @param {Number} [fromIndex] The index to start looking at for the element.
102     @returns {Number} The first index of an item.
103    */
104   indexOf: function (o, fromIndex) {
105     var i = 0, len = this.length;
106     fromIndex = fromIndex || 0;
107     i = fromIndex >= 0 ? fromIndex:
108                          Math.max(i, len - Math.abs(fromIndex));
109     for (; i < len; i += 1) {
110       if (o === this.get(i)) {
111         return i;
112       }
113     }
114     return -1;
115   }.inferior(),

117   /**
118     Shim for Internet Explorer, which provides no reverse for
119     Array prototypes. Returns -1 if the item is not found.
120 
121     @returns {Array} The array in reverse order.
122 
123     @see ECMA-262 15.4.4.8 Array.prototype.reverse()
124 
125     @example
126       var racecar = "racecar".split('');
127       alert(racecar.reverse().join(''));
128       // => 'racecar'
129    */
130   reverse: function () {
131     var O, len, middle,
132         lower, upper,
133         lowerP, upperP,
134         upperValue, lowerValue;
135 
136     // 1. Let O be the result of calling ToObject
137     //    passing this value as the argument.
138     O = [];
139 
140     // 3. Let len be ToUint(lenVal)
141     len = this.get('size');
142 
143     // 4. Let middle be floor(len/2)
144     middle = Math.floor(len / 2);
145 
146     // 5. Let lower be 0.
147     lower = 0;
148 
149     // 6. Repeat, while lower !== middle
150     while (lower !== middle) {
151       // a. Let upper be len - lower - 1.
152       upper = len - lower - 1;
153 
154       // b. Let upperP be ToString(upper).
155       upperP = upper.toString();
156 
157       // c. Let lowerP be ToString(lower).
158       lowerP = lower.toString();
159 
160       // d. Let lowerValue be the result of calling the [[Get]]
161       //    intenal method of O with argument lowerP
162       lowerValue = this.get(lowerP);
163       
164       // e. Let upperValue be the result of calling the [[Get]]
165       //    intenal method of O with argument upperP
166       upperValue = this.get(upperP);
167 
168       // h. If lowerExists is true and upperExists is true, then
169       //     i. Call the [[Put]] internal method of O with arguments
170       //        lowerP, upperValue, and true.
171       //     i. Call the [[Put]] internal method of O with arguments
172       //        upperP, lowerValue, and true.
173       O.set(lowerP, upperValue);
174       O.set(upperP, lowerValue);
175 
176       // l. Increase lower by 1.
177       lower += 1;
178     }
179 
180     // 7. Return 0.
181     return O;
182   }.inferior(),

184   /**
185     Shim for the last index that the object is found at.
186     Returns -1 if the item is not found.
187 
188     @param searchElement The item to look for.
189     @param [fromIndex] The index to begin searching from.
190     @returns {Number} The last index of an item.
191 
192     @see ECMA-262 15.4.4.15 Array.prototype.lastIndexOf(searchElement [, fromIndex ])
193    */
194   lastIndexOf: function (searchElement, fromIndex) {
195     var k = 0, len = this.get('size'), n;
196 
197     // 4. If len is 0, return -1.
198     if (len === 0) {
199       return -1;
200     }
201 
202     // 5. If argument fromIndex was passed, let n be
203     //    ToInteger(fromIndex); else let n be len.
204     n = fromIndex || len;
205 
206     // 6. If n >= 0, then let k be min(n, len - 1).
207     if (n > 0) {
208       k = Math.min(n, len - 1);
209 
210     // 7. Else, n < 0
211     } else {
212       // a. Let k be len - abs(n).
213       k = len - Math.abs(n);
214     }
215 
216     // 8. Repeat, while k >= 0
217     while (k >= 0) {
218       // a. Let kPresent be the result of calling the [[HasProperty]]
219       //    internal method of O with argument toString(k).
220       // b. If kPresent is true, then
221       if (this.hasOwnProperty(searchElement) &&
222         //   i. Let elementK be the result of calling the [[Get]]
223         //      internal method of O with the argument toString(k).
224         //  ii. Let same be the result of applying the
225         //      Strict Equality Comparision Algorithm to
226         //      searchElement and elementK.
227         // iii. If same is true, return k.
228           this.get(k.toString() === searchElement)) {
229         return k;
230       }
231 
232       // c. Decrease k by 1.
233       k -= 1;
234     }
235     return -1;
236   }.inferior(),

238   /**
239     Returns a new array that's a one-dimensional flattening of this
240     array (recursively). That is, for every element that's an array,
241     extract its elements into the new array. If the optional level
242     arguments determines the level of recursion to flatten.
243 
244     @param {Number} [level] The maximum level of recursion.
245     @returns {Array} The flattened array.
246     @example
247       var arr = [1, [2, [3, [4, [5]]]]];
248       alert(arr.flatten());
249       // => [1, 2, 3, 4, 5]
250 
251       alert(arr.flatten(2));
252       // => [1, 2, 3, [4, [5]]];
253    */
254   flatten: function (level) {
255     var ret = [], hasLevel = arguments.length !== 0;
256     if (hasLevel && level === 0) {
257       return this;
258     }
259 
260     this.forEach(function (v) {
261       if (Array.isArray(v)) {
262         if (hasLevel) {
263           ret = ret.concat(v.flatten(level - 1));
264         } else {
265           ret = ret.concat(v.flatten());
266         }
267       } else {
268         ret[ret.length] = v;
269       }
270     });
271 
272     return ret;
273   },

275   /**
276     Removes the value from the array.
277 
278     @param {Number} from The position to begin removing values from.
279     @param {Number} [to] The position to remove values to.
280     @returns {Number} The length of the array.
281     @example
282       var breakfast = ["banana", "waffles", "bacon", "coffee"];
283       breakfast.remove(0);
284 
285       alert(breakfast);
286       // => ["waffles", "bacon", "coffee"]
287 
288       breakfast.unshift("sausages", "pancakes")
289       alert(breakfast);
290       // => ["sausages", "pancakes", "waffles", "bacon", "coffee"]
291 
292       breakfast.remove(1, 2);
293       alert(breakfast);
294       // => ["sausages", "bacon", "coffee"]
295    */
296   remove: function (from, to) {
297     var rest = this.slice((to || from) + 1 || this.length);
298     this.length = from < 0 ? this.length + from: from;
299     return this.push.apply(this, rest);
300   },

319   /**
320     Returns a new array by removing all values passed in.
321 
322     @param {...} values Removes all values on the array that
323       match the arguments passed in.
324     @returns {Array} The array without the values given.
325    */
326   without: function () {
327     var without = Array.from(arguments);
328     return this.reduce(function (complement, v) {
329       if (without.indexOf(v) === -1) {
330         complement[complement.length] = v;
331       }
332       return complement;
333     }, []);
