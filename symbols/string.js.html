  6     Returns the string repeated the specified number of times.
  7 
  8     @param {Number} n The number of times to repeat this string.
  9     @param {String} [separator] The separator to put between each iteration of the string.
 10     @returns {String} The string repeated n times.
 11     @example
 12       alert("Stop hittin' yourself. ".repeat(50));
 13    */
 14   repeat: function (n, sep) {
 15     sep = sep || '';
 16     return n < 1 ? '': (new Array(n)).join(this + sep) + this;
 17   },

 32   /**
 33     Format formats a string in the vein of Python's format,
 34     Ruby #{templates}, and .NET String.Format.
 35 
 36     To write { or } in your Strings, just double them, and
 37     you'll end up with a single one.
 38 
 39     If you have more than one argument, then you can reference
 40     by the argument number (which is optional on a single argument).
 41 
 42     If you want to tie into this, and want to specify your own
 43     format specifier, override toFormat on your object, and it will
 44     pass you in the specifier (after the colon). You return the
 45     string it should look like, and that's it!
 46 
 47     For an example of an formatting extension, look at the Date mix.
 48     It implements the Ruby/Python formatting specification for Dates.
 49 
 50     @returns {String} The formatted string.
 51     @example
 52       alert("b{0}{0}a".format('an'));
 53       // => "banana"
 54 
 55     @example
 56       alert("I love {pi:.{precision}}".format({ pi: 22 / 7, precision: 2 }));
 57       // => "I love 3.14"
 58 
 59     @example
 60       alert("The {thing.name} is {thing.desc}.".format({
 61         thing: {
 62           name: 'cake',
 63           desc: 'a lie'
 64         }
 65       }));
 66       // => "The cake is a lie."
 67 
 68     @example
 69       alert(":-{{".format());  // Double {{ or }} to escape it.
 70       // => ":-{"
 71    */
 72   format: Espresso.inferior(function () {
 73     return Espresso.vformat(this, Espresso.A(arguments));
 74   }),

 76   /**
 77     Formatter for `String`s.
 78 
 79     Don't call this function- It's here for `Espresso.format`
 80     to take care of buisiness for you.
 81 
 82     @param {String} spec The specifier string.
 83     @returns {String} The string formatted using the format specifier.
 84    */
 85   toFormat: Espresso.inferior(function (spec) {
 86     var match = spec.match(Espresso.FORMAT_SPECIFIER),
 87         align = match[1],
 88         fill = match[2] || ' ',
 89         minWidth = match[6] || 0,
 90         maxWidth = match[7] || null, len, before, after, value,
 91         length = this.length;
 92 
 93     if (align) {
 94       align = align.slice(-1);
 95     }
 96 
 97     len = Math.max(minWidth, length);
 98     before = len - length;
 99     after = 0;
100 
101     switch (align) {
102     case '<':
103       after = before;
104       before = 0;
105       break;
106     case '^':
107       after = Math.ceil(before / 2);
108       before = Math.floor(before / 2);
109       break;
110     }
111 
112     value = this;
113     if (maxWidth != null) {
114       maxWidth = +maxWidth.slice(1);
115       value = isNaN(maxWidth) ? value : value.slice(0, maxWidth);
116     }
117 
118     return fill.repeat(before) + value + fill.repeat(after);
119   })
