 37     Refine allows for function-by-function refinements that
 38     reopens the function implementation without editing the
 39     original function's contents. With this, you can implement
 40     OO constructs like abstract base classes.
 41 
 42     Refinements to a function recieve a prepended argument to
 43     the argument list which is the original function that
 44     is being refined (if there isn't an original function that's
 45     being refined, a empty function will be provided for consistency).
 46 
 47     Calling the refined function should be done like so:
 48 
 49         Machiatto = mix({
 50           pull: function (original) {
 51             var espresso = original();
 52             return espresso + milk;
 53           }.refine()
 54         }).into(Espresso);
 55

 60           {@link Function#bind}, it will _not_ work.
 61 
 62     @returns {Function} The reciever.
 63    */

 72       if (!Espresso.isCallable(base)) {
 73         return value;
 74       }

 92   /**
 93     Currying transforms a function by transforming by composing it into
 94     functions that each take a subset of the arguments of the whole.
 95     The aggregate of all of the arguments passed into curried function
 96     and the actual function call is what will be provided to the function
 97     that's being curried.
 98 
 99     This effectively means that you can transform a very simple call into
100     atomic calls as follows:
101 
102         var mult = function () {
103           return Espresso.A(arguments).reduce(function (E, x) { return E * x; }, 1);
104         };
105 
106         alert(mult.curry(2, 2)());
107         // => 4
108 
109     Or specify a function that is a subset of the first:
110 
111         var add = function () {
112           return Espresso.A(arguments).reduce(function (E, x) { return E + x; }, 0);
113         };
114 
115         var inc = add.curry(1);
116 
117         alert(inc(5));
118         // => 6
119 
120     In layman's terms, `curry` will prepopulate arguments for a function.
121     @param {...} args Arguments that will be curried to the function.
122     @returns {Function} The function with the arguments provided saved for later.
123    */
124   curry: function () {
125     var Target, A;
126 
127     // 1. Let Target be the this value.
128     Target = this;
129

131     if (!Espresso.isCallable(Target)) {

133     }

136     //    all argument values (arg1, arg2, etc), in order
137     A = Espresso.A(arguments);
138

140       return Target.apply(this, A.concat(Espresso.A(arguments)));
141     };

144   /** @function
145     @desc
146     Bind the value of `this` on a function before hand,
147     with any extra arguments being passed in as initial
148     arguments.
149 
150     This implementation conforms to the ECMAScript 5
151     standard.
152 
153         var barista = function (tpl) {
154           alert(tpl.format(this));
155           return arguments.callee.bind(this, "Order up! Your {} is ready!");
156         };
157 
158         orderUp = barista.call("espresso", "I would like an {}");
159         // -> "I would like an espresso."
160 
161         orderUp();
162         // -> "Order up! Your espresso is ready!"
163

165     @returns {Function} The function passed in, wrapped to ensure `this`
166       is the correct scope.
167    */

169     var Target, A;
170 
171     // 1. Let Target be the this value.
172     Target = this;
173 
174     // 2. If IsCallable(Target) is false, throw a TypeError exception
175     if (!Espresso.isCallable(Target)) {
176       throw new TypeError("The Target is not callable.");
177     }
178 
179     // 3. Let A be a new (possibly empty) internal list of
180     //    all argument values provided after self
181     //    (arg1, arg2, etc), in order
182     A = Espresso.A(arguments).slice(1);
183 
184     var bound = function () {
185 
186       if (this instanceof bound) {
187         // 15.3.4.5.2 [[Construct]]
188         // When the [[Construct]] internal method of a function object, F,
189         // that was created using the bind function is called with a list of
190         // arguments ExtraArgs, the following steps are taken:
191 
192         // 1. Let the target be the value of F's [[TargetFunction]] internal property.
193         // 2. If target has no [[Construct]] internal method, a TypeError exception is thrown.
194         // 3. Let boundArgs be the value of F's [[BoundArgs]] internal property.
195         // 4. Left args be a new list containing the same values as the list boundArgs in the same order followed by the same values as the list ExtraArgs in the same order.
196         // 5. Return the result of calling the [[Construct]] internal method of target providing args as the arguments.
197         var Type = function () {}, that;
198         Type.prototype = Target.prototype;
199         that = new Type();
200 
201         Target.apply(that, A.concat(Espresso.A(arguments)));
202         return that;
203       } else {
204         // 15.3.4.5.1 [[Call]]
205         // When the [[Call]] internal method of a function object, F,
206         // which was created using the bind function is called with a this
207         // value and a list of arguments ExtraArgs, the following steps are taken:
208         // 1. Let boundArgs be the value of F's [[BoundArgs]] internal property.
209         // 2. Let boundThis be the value of F's [[BoundThis]] internal property.
210         // 3. Let target be the value of F's [[TargetFunction]] internal property.
211         return Target.apply(self, A.concat(Espresso.A(arguments)));
212       }
213     };
214     return bound;
215   })
